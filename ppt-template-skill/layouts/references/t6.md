## T6: 数据指标页

```javascript
function createKpiSlide(pres, C, data) {
  // data = { title, cards: [{ label, items: [{ name, value }] }], bottomText }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };

  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 5, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  const cards = data.cards;
  const n = cards.length;
  const gap = 0.4;
  const totalW = 9.0;
  const cardW = (totalW - gap * (n - 1)) / n;

  // 灰底
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 1.0, w: totalW, h: 3.4, fill: { color: C.primaryLight, transparency: 50 } });

  cards.forEach((card, i) => {
    const cx = 0.5 + i * (cardW + gap);
    slide.addShape(pres.shapes.RECTANGLE, { x: cx + 0.2, y: 1.2, w: 1.6, h: 0.38, fill: { color: C.primary } });
    slide.addText(card.label, {
      x: cx + 0.2, y: 1.2, w: 1.6, h: 0.38,
      fontSize: 12, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    card.items.forEach((item, j) => {
      const ry = 1.85 + j * 0.55;
      slide.addText(item.name, {
        x: cx + 0.2, y: ry, w: cardW * 0.55, h: 0.35,
        fontSize: 11, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
      });
      slide.addText(item.value, {
        x: cx + cardW * 0.5, y: ry, w: cardW * 0.45, h: 0.35,
        fontSize: 13, fontFace: FONT, color: C.textDark, bold: true, align: "right", valign: "middle", margin: 0
      });
      slide.addShape(pres.shapes.RECTANGLE, { x: cx + 0.2, y: ry + 0.38, w: cardW - 0.4, h: 0.008, fill: { color: C.border } });
    });
  });

  if (data.bottomText) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 4.55, w: totalW, h: 0.55, fill: { color: C.dark } });
    slide.addText(data.bottomText, {
      x: 0.5, y: 4.55, w: totalW, h: 0.55,
      fontSize: 16, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }
}
```
