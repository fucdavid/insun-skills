## T37: 银河M9评论区修复与传播拆解

适合：上市前评论区口碑修复、单链接维护成效、传播侧扩散数据复盘。

版式特征：评论区与传播侧左右双区，突出评论维护后的正面率圆环。

函数：`createM9CommentRepairSlide()`

依赖：先复制 `references/common.md` 中的 `FONT`、`addScreenshot()`，再复制本模板。所有颜色均为 6 位 HEX，不带 `#`。

> 注意：如果在同一个生成脚本中连续使用 T35-T43 多个 M9 模板，`m9Colors()`、`addM9Header()` 等基础辅助函数只保留一份，避免重复声明。单独使用本文件时可直接整段复制。

```javascript
// =====================================================
// GEELY / 银河 M9 蓝黑战报基础辅助函数
// 依赖 references/common.md: FONT, addScreenshot
// 画布：LAYOUT_16X9，10 × 5.625 inch
// 说明：同一 JS 文件中如果连续使用多个 M9 模板，基础辅助函数只保留一份即可。
// =====================================================

function m9Colors(C = {}) {
  return {
    blue: C.primary || "1F4BA8",
    blue2: "113D91",
    lightBlue: "EAF3F9",
    paleBlue: "DDEAF4",
    teal: "003B50",
    black: "000000",
    white: "FFFFFF",
    yellow: "FFF200",
    red: "E60012",
    green: "72B843",
    orange: "F28C28",
    grayText: C.textMid || "4D4D4D",
    darkText: C.textDark || "111111",
    line: C.border || "D8E2EA",
    card: C.cardBg || "FFFFFF",
    bg: C.bg || "FFFFFF",
    softGreen: "EAF4E5",
    softRed: "FDEBEC",
    softYellow: "FFF3D6",
    midGray: "BFBFBF",
    lightGray: "F3F5F7"
  };
}

function addM9TopCorner(slide, pres, C) {
  const G = m9Colors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0, y: 0, w: 0.16, h: 0.07,
    fill: { color: G.teal }, line: { color: G.teal }
  });
}

function addGeelyLogoText(slide, C) {
  const G = m9Colors(C);
  slide.addText("GEELY", {
    x: 8.95, y: 0.14, w: 0.9, h: 0.18,
    fontSize: 11, fontFace: FONT, color: "596775",
    bold: true, align: "right", margin: 0, breakLine: false
  });
}

function addM9RichText(slide, parts, box, baseOptions) {
  const runs = (parts || []).map(p => ({
    text: p.text,
    options: {
      color: p.color || baseOptions.color,
      bold: p.bold !== undefined ? p.bold : baseOptions.bold,
      fontFace: FONT
    }
  }));
  slide.addText(runs, Object.assign({}, baseOptions, box));
}

function addM9Header(slide, pres, C, data) {
  // data = { kicker, title, titleParts, subTitle }
  const G = m9Colors(C);
  slide.background = { color: G.bg };
  addM9TopCorner(slide, pres, C);
  addGeelyLogoText(slide, C);

  if (data.kicker) {
    slide.addText(`/ ${data.kicker} /`, {
      x: 2.2, y: 0.23, w: 5.6, h: 0.22,
      fontSize: 9, fontFace: FONT, color: G.darkText,
      align: "center", valign: "middle", margin: 0
    });
  }

  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0.25, y: 0.58, w: 9.5, h: 0.38,
    fill: { color: G.blue }, line: { color: G.blue }
  });
  if (data.titleParts && data.titleParts.length) {
    addM9RichText(slide, data.titleParts, { x: 0.35, y: 0.63, w: 9.3, h: 0.28 }, {
      fontSize: 12.5, fontFace: FONT, color: G.white, bold: true,
      align: "center", valign: "middle", margin: 0
    });
  } else {
    slide.addText(data.title || "", {
      x: 0.35, y: 0.63, w: 9.3, h: 0.28,
      fontSize: 12.5, fontFace: FONT, color: G.white,
      bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  if (data.subTitle) {
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 0.25, y: 1.02, w: 9.5, h: 0.19,
      fill: { color: G.black }, line: { color: G.black }
    });
    slide.addText(data.subTitle, {
      x: 0.35, y: 1.045, w: 9.3, h: 0.13,
      fontSize: 6.5, fontFace: FONT, color: G.white,
      bold: true, align: "center", valign: "middle", margin: 0
    });
  }
}

function addSectionBar(slide, pres, C, text, x, y, w, h, opts = {}) {
  const G = m9Colors(C);
  const fill = opts.fill || G.black;
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: fill }, line: { color: fill }
  });
  slide.addText(text, {
    x: x + 0.04, y: y + 0.03, w: w - 0.08, h: h - 0.06,
    fontSize: opts.fontSize || 8.5, fontFace: FONT,
    color: opts.color || G.white, bold: true,
    align: opts.align || "center", valign: "middle", margin: 0
  });
}

function addBlueNote(slide, pres, C, text, x, y, w, h) {
  const G = m9Colors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h, fill: { color: G.paleBlue }, line: { color: G.paleBlue }
  });
  slide.addText(text, {
    x: x + 0.08, y: y + 0.04, w: w - 0.16, h: h - 0.08,
    fontSize: 7.5, fontFace: FONT, color: G.red,
    bold: true, align: "center", valign: "middle", margin: 0
  });
}

function addMetricTable(slide, pres, C, metrics, x, y, w, h, opts = {}) {
  // metrics = [{label, value, color?}]
  const G = m9Colors(C);
  const n = metrics.length;
  const colW = w / n;
  for (let i = 0; i < n; i++) {
    const cx = x + i * colW;
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y, w: colW, h: h * 0.48,
      fill: { color: opts.headerFill || G.paleBlue },
      line: { color: G.white, width: 0.5 }
    });
    slide.addText(metrics[i].label, {
      x: cx + 0.03, y: y + 0.04, w: colW - 0.06, h: h * 0.38,
      fontSize: opts.labelSize || 7.5, fontFace: FONT, color: G.darkText,
      bold: true, align: "center", valign: "middle", margin: 0
    });
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: y + h * 0.48, w: colW, h: h * 0.52,
      fill: { color: opts.bodyFill || G.white },
      line: { color: G.white, width: 0.5 }
    });
    slide.addText(String(metrics[i].value), {
      x: cx + 0.03, y: y + h * 0.55, w: colW - 0.06, h: h * 0.36,
      fontSize: opts.valueSize || 9.5, fontFace: FONT,
      color: metrics[i].color || G.black, bold: !!metrics[i].bold,
      align: "center", valign: "middle", margin: 0
    });
  }
}

function addPlaceholder(slide, pres, C, label, x, y, w, h) {
  const G = m9Colors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: G.lightGray }, line: { color: G.line, width: 0.5 }
  });
  slide.addText(label || "截图占位", {
    x: x + 0.05, y: y + h / 2 - 0.1, w: w - 0.1, h: 0.2,
    fontSize: 7, fontFace: FONT, color: "8A8A8A",
    align: "center", valign: "middle", margin: 0
  });
}

async function addM9Image(slide, pres, C, img, x, y, w, h, label) {
  if (img && typeof img === "string") {
    await addScreenshot(slide, img, x, y, w, h);
  } else if (img && img.path) {
    await addScreenshot(slide, img.path, x, y, w, h);
  } else {
    addPlaceholder(slide, pres, C, label, x, y, w, h);
  }
}

async function addImageGrid(slide, pres, C, images, x, y, w, h, rows, cols, gap = 0.06) {
  const cellW = (w - gap * (cols - 1)) / cols;
  const cellH = (h - gap * (rows - 1)) / rows;
  let idx = 0;
  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      await addM9Image(slide, pres, C, images && images[idx], x + c * (cellW + gap), y + r * (cellH + gap), cellW, cellH, `截图 ${idx + 1}`);
      idx++;
    }
  }
}

function addSmallBulletLines(slide, C, lines, x, y, w, h, opts = {}) {
  const G = m9Colors(C);
  slide.addText(lines.join("\n"), {
    x, y, w, h,
    fontSize: opts.fontSize || 6.2, fontFace: FONT,
    color: opts.color || G.grayText,
    breakLine: false, fit: "shrink",
    margin: 0.02, valign: "top",
    lineSpacingMultiple: 1.05
  });
}

function addPercentBox(slide, pres, C, label, value, x, y, w, h, color) {
  const G = m9Colors(C);
  addSectionBar(slide, pres, C, label, x, y, w, 0.2);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y: y + 0.22, w, h: h - 0.22,
    fill: { color: G.white }, line: { color: G.white }
  });
  slide.addText(value, {
    x, y: y + 0.27, w, h: h - 0.32,
    fontSize: 16, fontFace: FONT, color,
    bold: true, align: "center", valign: "middle", margin: 0
  });
}

function addBottomBlackNote(slide, pres, C, text, x, y, w, h, opts = {}) {
  const G = m9Colors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h, fill: { color: G.black }, line: { color: G.black }
  });
  const parts = opts.parts;
  if (parts && parts.length) {
    addM9RichText(slide, parts, { x: x + 0.08, y: y + 0.05, w: w - 0.16, h: h - 0.1 }, {
      fontSize: opts.fontSize || 8, fontFace: FONT, color: G.white, bold: true,
      align: opts.align || "center", valign: "middle", margin: 0
    });
  } else {
    slide.addText(text, {
      x: x + 0.08, y: y + 0.05, w: w - 0.16, h: h - 0.1,
      fontSize: opts.fontSize || 8, fontFace: FONT,
      color: opts.color || G.white, bold: true,
      align: opts.align || "center", valign: "middle", margin: 0
    });
  }
}

// -----------------------------------------------------
// 3. 图3：评论区修复 / 拆解舆情双区
// -----------------------------------------------------
async function createM9CommentRepairSlide(pres, C, data) {
  // data = { kicker, title/titleParts, subTitle,
  //   leftMetrics, rightMetrics, leftTitle, rightTitle,
  //   leftImages, rightImages, centerPercent, commentSnippets, bottomNote }
  const slide = pres.addSlide();
  const G = m9Colors(C);
  addM9Header(slide, pres, C, data);

  addMetricTable(slide, pres, C, data.leftMetrics || [], 0.25, 1.28, 4.55, 0.52);
  addMetricTable(slide, pres, C, data.rightMetrics || [], 5.25, 1.28, 4.50, 0.52);

  addSectionBar(slide, pres, C, data.leftTitle || "评论区", 0.25, 1.92, 4.55, 0.20, { fill: G.blue, fontSize: 7 });
  addSectionBar(slide, pres, C, data.rightTitle || "传播侧", 5.25, 1.92, 4.50, 0.20, { fill: G.blue, fontSize: 7 });

  await addImageGrid(slide, pres, C, data.leftImages || [], 0.33, 2.22, 4.25, 2.05, 1, 2, 0.08);
  await addImageGrid(slide, pres, C, data.rightImages || [], 5.40, 2.22, 4.20, 2.05, 1, 2, 0.08);

  // 中央正面率圆环，以几何图形模拟
  const cx = 2.78;
  const cy = 3.70;
  slide.addShape(pres.shapes.OVAL, { x: cx, y: cy, w: 0.82, h: 0.82, fill: { color: G.white }, line: { color: G.green, width: 2 } });
  slide.addText(data.centerPercent || "80%\n正面率", {
    x: cx + 0.08, y: cy + 0.13, w: 0.66, h: 0.52,
    fontSize: 10, fontFace: FONT, color: G.green,
    bold: true, align: "center", valign: "middle", margin: 0,
    fit: "shrink"
  });

  if (data.commentSnippets && data.commentSnippets.length) {
    const snippets = data.commentSnippets.slice(0, 4);
    for (let i = 0; i < snippets.length; i++) {
      slide.addShape(pres.shapes.RECTANGLE, { x: 0.30, y: 4.35 + i * 0.18, w: 4.60, h: 0.15, fill: { color: i % 2 ? G.white : G.lightGray }, line: { color: G.white } });
      slide.addText(snippets[i], { x: 0.38, y: 4.37 + i * 0.18, w: 4.42, h: 0.11, fontSize: 4.6, fontFace: FONT, color: G.grayText, margin: 0, fit: "shrink" });
    }
  }
  addBottomBlackNote(slide, pres, C, data.bottomNote || "单链接做到500+评论、600+点赞，对维护80%正面率", 0.25, 5.08, 9.5, 0.34, { color: G.yellow, fontSize: 8.5 });
}
```

### 使用要点

1. `data.titleParts` 可用于标题中黄色高亮关键词；普通标题直接传 `data.title`。
2. 所有截图字段都通过 `addM9Image()` / `addImageGrid()` 间接调用 `addScreenshot()`，保持等比缩放。
3. 图片为空时会自动生成灰色占位框，方便先搭建页面结构。
