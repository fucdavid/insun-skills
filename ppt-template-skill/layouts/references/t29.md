## T29: 多列数据仪表板页（顶部摘要条 + N列编号分栏，每列有标题+摘要+多行数据条）

适合：APP活跃用户洞察、多维度数据分析、用户特征多列展示。每列一个维度（如职业特征、购车特征、网络特征、行为特征），列内展示多行数据条/进度条。

```javascript
function createDataDashboardSlide(pres, C, data) {
  // data = {
  //   title, summary,
  //   columns: [{
  //     number,    // 编号如"01"
  //     label,     // 维度名称如"职业特征"
  //     summary,   // 维度摘要（显示在标题色条上）
  //     items: [{ name, value, barWidth? }]  // 数据条列表，barWidth为0-1的比例值
  //   }],
  //   sourceNote   // 底部数据来源
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 摘要条
  if (data.summary) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.85, w: 9, h: 0.4, fill: { color: C.primary } });
    slide.addText(data.summary, {
      x: 0.7, y: 0.85, w: 8.6, h: 0.4,
      fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  const cols = data.columns || [];
  const n = cols.length;
  const gap = 0.15;
  const totalW = 9.0;
  const colW = (totalW - gap * (n - 1)) / n;
  const colStartY = 1.45;
  const colH = 3.75;

  cols.forEach((col, i) => {
    const cx = 0.5 + i * (colW + gap);

    // 编号圆圈 + 标题
    slide.addShape(pres.shapes.OVAL, {
      x: cx + 0.08, y: colStartY, w: 0.35, h: 0.35,
      fill: { color: C.primary }
    });
    slide.addText(col.number || String(i + 1).padStart(2, "0"), {
      x: cx + 0.08, y: colStartY, w: 0.35, h: 0.35,
      fontSize: 12, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    slide.addText(col.label || "", {
      x: cx + 0.5, y: colStartY, w: colW - 0.6, h: 0.35,
      fontSize: 13, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });

    // 维度摘要条
    if (col.summary) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx, y: colStartY + 0.42, w: colW, h: 0.35,
        fill: { color: C.primary }
      });
      slide.addText(col.summary, {
        x: cx + 0.08, y: colStartY + 0.42, w: colW - 0.16, h: 0.35,
        fontSize: 8, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
      });
    }

    // 数据条区域
    const items = col.items || [];
    const barStartY = colStartY + 0.9;
    const barAreaH = colH - 1.0;
    const barH = Math.min(0.45, barAreaH / Math.max(items.length, 1));

    items.forEach((item, j) => {
      const by = barStartY + j * barH;
      // 标签
      slide.addText(item.name || "", {
        x: cx + 0.08, y: by, w: colW * 0.55, h: barH * 0.5,
        fontSize: 8, fontFace: FONT, color: C.textDark, align: "left", valign: "middle", margin: 0
      });
      // 数据条背景
      const barX = cx + 0.08;
      const barW = colW - 0.16;
      const barY2 = by + barH * 0.5;
      slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
        x: barX, y: barY2, w: barW, h: barH * 0.35,
        fill: { color: C.border }, rectRadius: 0.03
      });
      // 数据条填充
      const fillW = (item.barWidth || 0.5) * barW;
      if (fillW > 0) {
        slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
          x: barX, y: barY2, w: fillW, h: barH * 0.35,
          fill: { color: C.primary }, rectRadius: 0.03
        });
      }
      // 数值标签
      slide.addText(item.value || "", {
        x: cx + colW * 0.6, y: by, w: colW * 0.35, h: barH * 0.5,
        fontSize: 9, fontFace: FONT, color: C.textDark, bold: true, align: "right", valign: "middle", margin: 0
      });
    });
  });

  // 底部数据来源
  if (data.sourceNote) {
    slide.addText(data.sourceNote, {
      x: 0.5, y: 5.35, w: 9, h: 0.2,
      fontSize: 7, fontFace: FONT, color: C.textLight, align: "left", valign: "middle", margin: 0
    });
  }
}
```
