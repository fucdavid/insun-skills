# 三角辐射枢纽

## T25: 三角辐射枢纽页（深色底 + 三个角点卡片汇聚到中心品牌枢纽 + 左侧图例说明）

适合：用户分层运营策略、三类人群×中心品牌的辐射结构、三方力量汇聚到一个核心。区别于 T12 的圆形多节点环绕，T25 强调"三角顶点 → 中心枢纽"的金字塔汇聚关系，三个卡片各有独立强调色，视觉冲击力更强。

适用场景示例：
- 三类用户分层 + 品牌APP汇聚点
- 三大业务支柱 + 集团核心战略
- 三方资源（产品/渠道/用户）+ 品牌中心

```javascript
function createTriHubSlide(pres, C, data) {
  // data = {
  //   title, subtitle,
  //   hub: { name, sub, tagline },
  //   corners: [
  //     { badge, title, subtitle, coreNeed, row1Label, row1Text, row2Label, row2Text, accentColor? },
  //     ... 共3个，顺序：[顶部, 左下, 右下]
  //   ],
  //   legend: { title, items: [{ label, desc }] },  // 可选，左上角图例区
  //   quote,        // 可选，图例区下方斜体引言
  //   bottomNote    // 可选，底部总结条
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  // 顶部主色细条
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 左上角小装饰L角
  slide.addShape(pres.shapes.LINE, {
    x: 0.3, y: 0.22, w: 0.18, h: 0,
    line: { color: C.primary, width: 1.5 }
  });
  slide.addShape(pres.shapes.LINE, {
    x: 0.3, y: 0.22, w: 0, h: 0.18,
    line: { color: C.primary, width: 1.5 }
  });

  // 主标题
  slide.addText(data.title, {
    x: 0.5, y: 0.35, w: 7, h: 0.5,
    fontSize: 22, fontFace: FONT, color: "FFFFFF", bold: true, align: "left", valign: "middle", margin: 0
  });
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 0.5, y: 0.85, w: 8, h: 0.3,
      fontSize: 11, fontFace: FONT, color: C.textLight, align: "left", valign: "middle", margin: 0
    });
  }

  // ========== 布局参数 ==========
  const SW = 10, hubR = 0.55;
  const CX = SW / 2, CY = 3.4;

  const cardW = 2.95, cardH = 1.5;
  const apex = { x: CX - cardW / 2, y: 1.35 };
  const bl   = { x: 0.4,            y: 4.0 };
  const br   = { x: SW - 0.4 - cardW, y: 4.0 };

  // 三个角点的强调色（默认从 C 调色板取，data.corners[i].accentColor 可覆盖）
  const defaultAccents = [C.primary, C.textMid, C.primaryLight];

  // ========== 连接线（先画，让卡片和枢纽盖在上面）==========
  const apexAnchor = { x: apex.x + cardW / 2, y: apex.y + cardH };
  const blAnchor   = { x: bl.x + cardW,       y: bl.y + 0.3 };
  const brAnchor   = { x: br.x,               y: br.y + 0.3 };

  function drawConnector(from, to, color) {
    const dx = to.x - from.x, dy = to.y - from.y;
    const d = Math.sqrt(dx * dx + dy * dy);
    const ux = dx / d, uy = dy / d;
    const endX = to.x - ux * hubR;
    const endY = to.y - uy * hubR;

    slide.addShape(pres.shapes.LINE, {
      x: from.x, y: from.y,
      w: endX - from.x, h: endY - from.y,
      line: { color: color, width: 1.5, transparency: 30 }
    });
    // 靠近枢纽的箭头点
    slide.addShape(pres.shapes.OVAL, {
      x: endX - 0.05, y: endY - 0.05, w: 0.1, h: 0.1,
      fill: { color: color }, line: { color: color, width: 0 }
    });
    // 卡片边缘的起点
    slide.addShape(pres.shapes.OVAL, {
      x: from.x - 0.04, y: from.y - 0.04, w: 0.08, h: 0.08,
      fill: { color: color }, line: { color: color, width: 0 }
    });
  }

  const accents = (data.corners || []).map((c, i) => c.accentColor || defaultAccents[i] || C.primary);
  drawConnector(apexAnchor, { x: CX, y: CY }, accents[0]);
  drawConnector(blAnchor,   { x: CX, y: CY }, accents[1]);
  drawConnector(brAnchor,   { x: CX, y: CY }, accents[2]);

  // ========== 左侧图例区（可选）==========
  if (data.legend) {
    const legX = 0.4, legY = 1.55, legW = 2.3;
    slide.addText(data.legend.title || "运行机制", {
      x: legX, y: legY, w: legW, h: 0.3,
      fontSize: 12, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });

    const items = data.legend.items || [];
    items.forEach((it, i) => {
      const iy = legY + 0.4 + i * 0.55;
      // 编号小圆
      slide.addShape(pres.shapes.OVAL, {
        x: legX, y: iy, w: 0.26, h: 0.26,
        fill: { color: C.dark }, line: { color: C.primary, width: 1 }
      });
      slide.addText(String(i + 1), {
        x: legX, y: iy, w: 0.26, h: 0.26,
        fontSize: 9, fontFace: FONT, color: C.primary, bold: true, align: "center", valign: "middle", margin: 0
      });
      // 标题
      slide.addText(it.label, {
        x: legX + 0.36, y: iy - 0.02, w: legW - 0.4, h: 0.22,
        fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "left", valign: "middle", margin: 0
      });
      // 描述
      slide.addText(it.desc || "", {
        x: legX + 0.36, y: iy + 0.18, w: legW - 0.4, h: 0.2,
        fontSize: 8, fontFace: FONT, color: C.textLight, align: "left", valign: "middle", margin: 0
      });
    });

    // 引言（可选） - 紧凑显示，避免与左下角卡片重叠
    if (data.quote) {
      const qy = legY + 0.4 + items.length * 0.55 + 0.05;
      slide.addShape(pres.shapes.LINE, {
        x: legX, y: qy, w: legW - 0.3, h: 0,
        line: { color: C.primary, width: 0.5, transparency: 50 }
      });
      slide.addText('"' + data.quote + '"', {
        x: legX, y: qy + 0.06, w: legW, h: 0.35,
        fontSize: 8, fontFace: FONT, color: C.primary, italic: true, align: "left", valign: "top",
        lineSpacingMultiple: 1.3, margin: 0
      });
    }
  }

  // ========== 中心品牌枢纽 ==========
  // 外圈描边
  slide.addShape(pres.shapes.OVAL, {
    x: CX - hubR - 0.08, y: CY - hubR - 0.08,
    w: (hubR + 0.08) * 2, h: (hubR + 0.08) * 2,
    fill: { color: C.dark }, line: { color: C.primary, width: 1 }
  });
  // 主圆
  slide.addShape(pres.shapes.OVAL, {
    x: CX - hubR, y: CY - hubR,
    w: hubR * 2, h: hubR * 2,
    fill: { color: C.primary }, line: { color: C.primary, width: 0 }
  });
  // 枢纽名称
  slide.addText(data.hub.name, {
    x: CX - hubR, y: CY - 0.32, w: hubR * 2, h: 0.32,
    fontSize: 14, fontFace: FONT, color: C.dark, bold: true, align: "center", valign: "middle", margin: 0
  });
  // 副标识
  if (data.hub.sub) {
    slide.addText(data.hub.sub, {
      x: CX - hubR, y: CY - 0.02, w: hubR * 2, h: 0.22,
      fontSize: 9, fontFace: FONT, color: C.dark, bold: true, align: "center", valign: "middle",
      charSpacing: 4, margin: 0
    });
  }
  // 标签语
  if (data.hub.tagline) {
    slide.addText(data.hub.tagline, {
      x: CX - hubR, y: CY + 0.22, w: hubR * 2, h: 0.2,
      fontSize: 8, fontFace: FONT, color: C.dark, italic: true, align: "center", valign: "middle", margin: 0
    });
  }

  // ========== 三个角卡片 ==========
  function drawCornerCard(pos, accent, c, idx) {
    const { x, y } = pos;
    // 卡片底
    slide.addShape(pres.shapes.RECTANGLE, {
      x, y, w: cardW, h: cardH,
      fill: { color: C.bg }, line: { color: C.border, width: 0.5 }
    });
    // 顶部强调色条
    slide.addShape(pres.shapes.RECTANGLE, {
      x, y, w: cardW, h: 0.05,
      fill: { color: accent }
    });
    // 左上角徽章
    slide.addText(c.badge, {
      x: x + 0.18, y: y + 0.1, w: cardW - 1.25, h: 0.2,
      fontSize: 7, fontFace: FONT, color: accent, bold: true, charSpacing: 2,
      align: "left", valign: "middle", margin: 0
    });
    // 右上角"核心需求"小圆角标签
    const pillW = 0.85, pillH = 0.24;
    slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
      x: x + cardW - 0.18 - pillW, y: y + 0.08, w: pillW, h: pillH,
      fill: { color: accent, transparency: 75 },
      line: { color: accent, width: 0.5 }, rectRadius: 0.05
    });
    slide.addText(c.coreNeed, {
      x: x + cardW - 0.18 - pillW, y: y + 0.08, w: pillW, h: pillH,
      fontSize: 9, fontFace: FONT, color: accent, bold: true, align: "center", valign: "middle", margin: 0
    });
    // 主标题
    slide.addText(c.title, {
      x: x + 0.18, y: y + 0.32, w: cardW - 0.36, h: 0.32,
      fontSize: 14, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });
    // 副标题
    slide.addText(c.subtitle, {
      x: x + 0.18, y: y + 0.62, w: cardW - 0.36, h: 0.22,
      fontSize: 8.5, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
    // 分隔线
    slide.addShape(pres.shapes.RECTANGLE, {
      x: x + 0.18, y: y + 0.88, w: cardW - 0.36, h: 0.008, fill: { color: C.border }
    });
    // 第一行（玩法/标签）
    slide.addShape(pres.shapes.OVAL, {
      x: x + 0.2, y: y + 0.97, w: 0.16, h: 0.16,
      fill: { color: accent }
    });
    slide.addText([
      { text: (c.row1Label || "玩法") + "  ", options: { color: accent, bold: true } },
      { text: c.row1Text, options: { color: C.textDark } }
    ], {
      x: x + 0.42, y: y + 0.93, w: cardW - 0.55, h: 0.24,
      fontSize: 8.5, fontFace: FONT, align: "left", valign: "middle", margin: 0
    });
    // 第二行（价值/标签）
    slide.addShape(pres.shapes.OVAL, {
      x: x + 0.2, y: y + 1.22, w: 0.16, h: 0.16,
      fill: { color: accent }
    });
    slide.addText([
      { text: (c.row2Label || "价值") + "  ", options: { color: accent, bold: true } },
      { text: c.row2Text, options: { color: C.textDark } }
    ], {
      x: x + 0.42, y: y + 1.18, w: cardW - 0.55, h: 0.24,
      fontSize: 8.5, fontFace: FONT, align: "left", valign: "middle", margin: 0
    });
  }

  const corners = data.corners || [];
  if (corners[0]) drawCornerCard(apex, accents[0], corners[0], 0);
  if (corners[1]) drawCornerCard(bl,   accents[1], corners[1], 1);
  if (corners[2]) drawCornerCard(br,   accents[2], corners[2], 2);

  // ========== 底部总结条（可选）==========
  if (data.bottomNote) {
    slide.addShape(pres.shapes.LINE, {
      x: 0.5, y: 5.42, w: 9, h: 0,
      line: { color: C.primary, width: 0.5, transparency: 40 }
    });
    slide.addText(data.bottomNote, {
      x: 0.5, y: 5.45, w: 9, h: 0.16,
      fontSize: 8, fontFace: FONT, color: C.textLight, align: "center", valign: "middle", margin: 0
    });
  }
}
```
