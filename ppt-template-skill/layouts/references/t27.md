## T27: 品牌对标调研页（品牌标识+策略描述+下方多栏截图展示）

适合：行业对标APP/小程序调研、竞品功能拆解。顶部展示品牌名称和核心模式，中间是策略描述段落，下方分栏展示截图或功能模块（每栏有标题+截图占位区+底部说明）。

```javascript
async function createBrandBenchmarkSlide(pres, C, data) {
  // data = {
  //   title,
  //   brand: { name, model, insight },   // 品牌名、模式名称、启示标签
  //   description,                       // 策略描述段落
  //   columns: [{ title, desc, images[] }],  // 下方分栏（2-4栏），每栏可含截图
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 7, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 品牌信息区
  const brandY = 0.85;
  // 品牌名称
  slide.addText(data.brand.name || "", {
    x: 0.5, y: brandY, w: 1.2, h: 0.7,
    fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
  });
  // 模式名称（大字强调）
  slide.addText(data.brand.model || "", {
    x: 1.8, y: brandY, w: 3, h: 0.35,
    fontSize: 16, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
  });
  // 启示标签
  if (data.brand.insight) {
    slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
      x: 5.0, y: brandY + 0.02, w: 4.5, h: 0.32,
      fill: { color: C.primary }, rectRadius: 0.05
    });
    slide.addText(data.brand.insight, {
      x: 5.0, y: brandY + 0.02, w: 4.5, h: 0.32,
      fontSize: 9, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  // 策略描述
  if (data.description) {
    slide.addText(data.description, {
      x: 1.8, y: brandY + 0.4, w: 7.7, h: 0.5,
      fontSize: 9, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
      lineSpacingMultiple: 1.4, margin: 0
    });
  }

  // 下方分栏
  const cols = data.columns || [];
  const n = cols.length;
  const colGap = 0.2;
  const colStartY = 1.95;
  const colW = (9.0 - colGap * (n - 1)) / n;
  const colH = 5.625 - colStartY - 0.15;

  for (let i = 0; i < n; i++) {
    const cx = 0.5 + i * (colW + colGap);
    const col = cols[i];

    // 栏标题条
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: colStartY, w: colW, h: 0.35,
      fill: { color: C.dark }
    });
    slide.addText(col.title, {
      x: cx, y: colStartY, w: colW, h: 0.35,
      fontSize: 11, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 截图区域（如果有图片则放置，否则用占位色块）
    const imgAreaY = colStartY + 0.4;
    const imgAreaH = colH - 0.85;

    if (col.images && col.images.length > 0) {
      const imgH = imgAreaH / col.images.length - 0.05;
      for (let j = 0; j < col.images.length; j++) {
        await addScreenshot(slide, col.images[j], cx + 0.05, imgAreaY + j * (imgH + 0.05), colW - 0.1, imgH);
      }
    } else {
      // 占位色块
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx + 0.05, y: imgAreaY, w: colW - 0.1, h: imgAreaH,
        fill: { color: C.primaryLight, transparency: 50 }
      });
    }

    // 底部说明
    if (col.desc) {
      const descY = colStartY + colH - 0.4;
      slide.addText(col.desc, {
        x: cx + 0.1, y: descY, w: colW - 0.2, h: 0.38,
        fontSize: 8, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
      });
    }
  }
}
```
