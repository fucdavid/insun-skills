## T3: 章节过渡页

```javascript
function createTransitionSlide(pres, C, data) {
  // data = { topText, mainText, bottomText }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  if (data.topText) {
    slide.addText(data.topText, {
      x: 1, y: 1.3, w: 8, h: 0.5,
      fontSize: 15, fontFace: FONT, color: C.textLight, align: "center", margin: 0
    });
  }
  slide.addText(data.mainText, {
    x: 0.5, y: 2.0, w: 9, h: 1.0,
    fontSize: 32, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", margin: 0
  });
  // 下方装饰线
  slide.addShape(pres.shapes.RECTANGLE, { x: 4.3, y: 3.2, w: 1.4, h: 0.03, fill: { color: C.primary } });
  if (data.bottomText) {
    slide.addText(data.bottomText, {
      x: 1, y: 3.5, w: 8, h: 0.7,
      fontSize: 14, fontFace: FONT, color: C.primary, align: "center", lineSpacingMultiple: 1.5, margin: 0
    });
  }
}
```
