## T26: 上下双行对照页（上行N个要点卡片 + 圆点连接 + 下行N个提炼卡片 + 底部总结条）

适合：课题解读与提炼、问题与对策对照、需求到方案的映射关系。上行展示原始要点，下行展示对应的提炼/归纳，圆点连接表示映射关系。

```javascript
function createDualRowMappingSlide(pres, C, data) {
  // data = {
  //   title, summary,
  //   topLabel,    // 上行区域标签，如"核心课题"
  //   bottomLabel, // 下行区域标签，如"课题提炼"
  //   topCards: [{ title, desc }],     // 上行卡片（2-4个）
  //   bottomCards: [{ title, desc, highlight }], // 下行卡片（与上行一一对应或独立）
  //   bottomNote   // 底部总结条
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.summary) {
    slide.addText(data.summary, {
      x: 0.75, y: 0.8, w: 8.5, h: 0.3,
      fontSize: 10, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  }

  const topCards = data.topCards || [];
  const bottomCards = data.bottomCards || [];
  const n = Math.max(topCards.length, bottomCards.length, 1);
  const gap = 0.2;
  const labelW = 0.7;
  const cardAreaX = 0.5 + labelW + 0.15;
  const cardAreaW = 9.0 - labelW - 0.15;
  const cardW = (cardAreaW - gap * (n - 1)) / n;

  const topY = 1.2;
  const topH = 1.35;
  const dotY = topY + topH + 0.08;
  const bottomY = dotY + 0.35;
  const bottomH = 1.35;

  // 上行标签
  if (data.topLabel) {
    slide.addText(data.topLabel, {
      x: 0.5, y: topY + topH / 2 - 0.2, w: labelW, h: 0.4,
      fontSize: 10, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  // 上行卡片
  topCards.forEach((card, i) => {
    const cx = cardAreaX + i * (cardW + gap);
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: topY, w: cardW, h: topH,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    slide.addText(card.title, {
      x: cx + 0.15, y: topY + 0.1, w: cardW - 0.3, h: 0.3,
      fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });
    slide.addText(card.desc || "", {
      x: cx + 0.15, y: topY + 0.45, w: cardW - 0.3, h: topH - 0.55,
      fontSize: 8, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
      lineSpacingMultiple: 1.4, margin: 0
    });
  });

  // 中间连接圆点
  topCards.forEach((_, i) => {
    const cx = cardAreaX + i * (cardW + gap) + cardW / 2;
    slide.addShape(pres.shapes.OVAL, {
      x: cx - 0.12, y: dotY, w: 0.24, h: 0.24,
      fill: { color: C.primary }
    });
  });

  // 下行标签
  if (data.bottomLabel) {
    slide.addText(data.bottomLabel, {
      x: 0.5, y: bottomY + bottomH / 2 - 0.2, w: labelW, h: 0.4,
      fontSize: 10, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  // 下行卡片
  bottomCards.forEach((card, i) => {
    const cx = cardAreaX + i * (cardW + gap);
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: bottomY, w: cardW, h: bottomH,
      fill: { color: C.primaryLight, transparency: 30 }
    });
    slide.addText(card.title, {
      x: cx + 0.15, y: bottomY + 0.1, w: cardW - 0.3, h: 0.3,
      fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });
    slide.addText(card.desc || "", {
      x: cx + 0.15, y: bottomY + 0.45, w: cardW - 0.3, h: bottomH - 0.75,
      fontSize: 9, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
    // 高亮关键词
    if (card.highlight) {
      slide.addText(card.highlight, {
        x: cx + 0.15, y: bottomY + bottomH - 0.35, w: cardW - 0.3, h: 0.25,
        fontSize: 12, fontFace: FONT, color: C.primary, bold: true, align: "center", valign: "middle", margin: 0
      });
    }
  });

  // 底部总结条
  if (data.bottomNote) {
    const noteY = bottomY + bottomH + 0.2;
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 0.5, y: noteY, w: 9.0, h: 0.45,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    slide.addText(data.bottomNote, {
      x: 0.7, y: noteY, w: 8.6, h: 0.45,
      fontSize: 10, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
  }
}
```
