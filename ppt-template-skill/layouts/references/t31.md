## T31: 策略执行矩阵页（左侧行标签 + 顶部列标签 + 单元格内容填充 + 底部汇总条）

适合：直播规划方案、分阶段策略执行计划、OTA传播矩阵、活动运营方案总表。左侧一列为维度标签（如目标/策略/执行线/费用），顶部为阶段或场景列头，交叉单元格填充具体内容。底部可加汇总条（如费用合计）。

```javascript
function createStrategyMatrixSlide(pres, C, data) {
  // data = {
  //   title,
  //   subtitle,         // 副标题/slogan（可选）
  //   colHeaders: [{ label, desc? }],   // 顶部列标签（2-4列）
  //   rowLabels: ["目标", "策略", ...],  // 左侧行标签
  //   cells: [           // 二维数组 cells[row][col]，每个cell是 { title?, text }
  //     [{ title: "xxx", text: "详细内容" }, { text: "..." }],  // 第1行的各列
  //     [{ text: "..." }, { text: "..." }],                     // 第2行的各列
  //   ],
  //   bottomBar,         // 底部汇总条文字（可选）
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.15, w: 0.06, h: 0.4, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.15, w: 7, h: 0.4,
    fontSize: 18, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 0.75, y: 0.55, w: 8.5, h: 0.25,
      fontSize: 10, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });
  }

  // 布局参数
  const labelW = 0.8;        // 左侧行标签列宽
  const startX = 0.4;
  const startY = 0.9;
  const totalW = 9.2;
  const colHeaders = data.colHeaders || [];
  const rowLabels = data.rowLabels || [];
  const cells = data.cells || [];
  const nCols = colHeaders.length;
  const nRows = rowLabels.length;
  const colGap = 0.08;
  const rowGap = 0.06;

  const contentW = totalW - labelW - 0.1;
  const colW = (contentW - colGap * (nCols - 1)) / nCols;
  const headerH = 0.45;

  // 预留底部汇总条空间
  const bottomBarH = data.bottomBar ? 0.4 : 0;
  const availH = 5.625 - startY - headerH - rowGap - bottomBarH - 0.15;
  const rowH = (availH - rowGap * (nRows - 1)) / nRows;
  const colStartX = startX + labelW + 0.1;

  // ===== 列标题头 =====
  colHeaders.forEach((h, ci) => {
    const cx = colStartX + ci * (colW + colGap);
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: startY, w: colW, h: headerH,
      fill: { color: C.primary }
    });
    const headerText = h.desc ? h.label + "\n" + h.desc : h.label;
    slide.addText(headerText, {
      x: cx + 0.08, y: startY, w: colW - 0.16, h: headerH,
      fontSize: h.desc ? 9 : 11, fontFace: FONT, color: "FFFFFF", bold: true,
      align: "center", valign: "middle", lineSpacingMultiple: 1.2, margin: 0
    });
  });

  // ===== 行标签 + 单元格 =====
  const cellStartY = startY + headerH + rowGap;

  rowLabels.forEach((label, ri) => {
    const ry = cellStartY + ri * (rowH + rowGap);

    // 行标签（深色背景）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: startX, y: ry, w: labelW, h: rowH,
      fill: { color: C.dark }
    });
    slide.addText(label, {
      x: startX, y: ry, w: labelW, h: rowH,
      fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true,
      align: "center", valign: "middle", margin: 0
    });

    // 该行的各列单元格
    const rowCells = cells[ri] || [];
    rowCells.forEach((cell, ci) => {
      if (ci >= nCols) return;
      const cx = colStartX + ci * (colW + colGap);

      // 单元格背景
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx, y: ry, w: colW, h: rowH,
        fill: { color: C.cardBg },
        line: { color: C.border, width: 0.5 }
      });

      // 单元格内标题（可选，用主色加粗）
      let textY = ry + 0.06;
      if (cell.title) {
        slide.addText(cell.title, {
          x: cx + 0.1, y: textY, w: colW - 0.2, h: 0.25,
          fontSize: 10, fontFace: FONT, color: C.primary, bold: true,
          align: "left", valign: "middle", margin: 0
        });
        textY += 0.28;
      }

      // 单元格正文
      if (cell.text) {
        slide.addText(cell.text, {
          x: cx + 0.1, y: textY, w: colW - 0.2, h: ry + rowH - textY - 0.06,
          fontSize: 8, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
          lineSpacingMultiple: 1.3, margin: 0
        });
      }
    });
  });

  // ===== 底部汇总条 =====
  if (data.bottomBar) {
    const barY = cellStartY + nRows * (rowH + rowGap);
    slide.addShape(pres.shapes.RECTANGLE, {
      x: startX, y: barY, w: totalW, h: bottomBarH,
      fill: { color: C.primaryLight }
    });
    slide.addText(data.bottomBar, {
      x: startX + 0.15, y: barY, w: totalW - 0.3, h: bottomBarH,
      fontSize: 9, fontFace: FONT, color: C.textDark, bold: true,
      align: "center", valign: "middle", margin: 0
    });
  }
}
```
