## T10: 暗底侧边目录页（深色背景 + 左侧大号竖排CONTENTS + 右侧带高亮的章节列表）

适合：多章节方案的目录页/章节切换页，当前章节用主色高亮+三角指示

```javascript
function createDarkSidebarTocSlide(pres, C, data) {
  // data = { items: ["章节1", ...], activeIndex: 0, bgImage? }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  // 可选右半背景图
  if (data.bgImage) {
    slide.addImage({ path: require("path").resolve(data.bgImage), x: 5.0, y: 0, w: 5.0, h: 5.625 });
    slide.addShape(pres.shapes.RECTANGLE, { x: 5.0, y: 0, w: 5.0, h: 5.625, fill: { color: C.dark, transparency: 30 } });
  }

  // 左侧深色面板
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 5.0, h: 5.625, fill: { color: C.dark } });

  // 底部主色条
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 5.35, w: 10, h: 0.04, fill: { color: C.primary } });

  // 竖排大字 "CONTENTS" 装饰（用半透明大字做背景）
  const letters = "CONTENTS".split("");
  letters.forEach((ch, i) => {
    slide.addText(ch, {
      x: 0.15, y: 0.3 + i * 0.6, w: 0.6, h: 0.55,
      fontSize: 32, fontFace: FONT, color: C.textLight, bold: true, align: "center", valign: "middle", margin: 0,
      transparency: 70
    });
  });

  // 章节列表
  const startY = 0.5;
  const lineH = 0.55;
  data.items.forEach((item, i) => {
    const isActive = (i === data.activeIndex);
    const iy = startY + i * lineH;

    // 高亮三角指示
    if (isActive) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: 1.3, y: iy + 0.08, w: 0.2, h: 0.2, fill: { color: C.primary }, rotate: 45
      });
    }

    // 左竖色条
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 1.65, y: iy, w: 0.05, h: 0.38,
      fill: { color: isActive ? C.primary : C.textLight }
    });

    // 文字
    slide.addText(item, {
      x: 1.9, y: iy, w: 2.8, h: 0.4,
      fontSize: isActive ? 16 : 14, fontFace: FONT,
      color: isActive ? C.primary : "FFFFFF",
      bold: isActive, align: "left", valign: "middle", margin: 0
    });
  });
}
```
