## T13: 全屏金句宣言页（深色/图片背景 + 上方引导文 + 超大主句 + 下方解释）

适合：愿景宣言、战略定调、精神口号、章节核心观点突出。比T3更有气势。

```javascript
function createVisionStatementSlide(pres, C, data) {
  // data = { topText, mainText, bottomText, highlightColor?, bgImage? }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  if (data.bgImage) {
    slide.addImage({ path: require("path").resolve(data.bgImage), x: 0, y: 0, w: 10, h: 5.625 });
    slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 5.625, fill: { color: "000000", transparency: 40 } });
  }

  // 上方引导语（较小字，偏灰）
  if (data.topText) {
    slide.addText(data.topText, {
      x: 0.8, y: 1.0, w: 8.4, h: 0.5,
      fontSize: 14, fontFace: FONT, color: C.textLight, align: "center", margin: 0
    });
  }

  // 左上角装饰斜杠
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.6, y: 1.7, w: 0.4, h: 0.06, fill: { color: C.primary }, rotate: -30 });

  // 核心主句（超大字，可包含高亮色文字）
  const hc = data.highlightColor || C.primary;
  slide.addText(data.mainText, {
    x: 0.5, y: 1.9, w: 9.0, h: 1.4,
    fontSize: 36, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
  });

  // 下方解释文字（主色，中等大小）
  if (data.bottomText) {
    slide.addText(data.bottomText, {
      x: 0.8, y: 3.5, w: 8.4, h: 1.0,
      fontSize: 16, fontFace: FONT, color: C.primary, bold: true, align: "center", valign: "top",
      lineSpacingMultiple: 1.6, margin: 0
    });
  }
}
```
