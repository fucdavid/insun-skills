## T45: BYD核心优势三栏洞察页

适合：背景洞察、产品优势归纳、三大卖点并列说明。

版式特征：顶部洞察结论 + 三列红黑卡片 + 大图 + 底部结论短句。

函数：`createBydCoreValueInsightSlide()`

依赖：先复制 `references/common.md` 中的 `FONT`、`addScreenshot()`，再复制本模板。所有颜色均为 6 位 HEX，不带 `#`。

> 注意：如果在同一个生成脚本中连续使用 T45-T54 多个 BYD 模板，`bydColors()`、`addBydHeader()` 等基础辅助函数只保留一份，避免重复声明。单独使用本文件时可直接整段复制。

```javascript
// =====================================================
// BYD 红黑策略页基础辅助函数
// 依赖 references/common.md: FONT, addScreenshot
// 画布：LAYOUT_16X9，10 × 5.625 inch
// 说明：同一 JS 文件中如果连续使用多个 BYD 模板，基础辅助函数只保留一份即可。
// =====================================================

function bydColors(C = {}) {
  return {
    red: C.primary || "C00000",
    red2: "D61A1A",
    black: C.dark || "111111",
    white: "FFFFFF",
    bg: C.bg || "FFFFFF",
    card: C.cardBg || "FFFFFF",
    line: C.border || "D9D9D9",
    softGray: "F2F2F2",
    lightGray: "F7F7F7",
    text: C.textDark || "111111",
    mid: C.textMid || "666666",
    yellow: "F4B400",
    paleYellow: "FFF2CC",
    beige: "F8EDD5"
  };
}

function addBydRichText(slide, parts, box, baseOptions) {
  const runs = (parts || []).map(p => ({
    text: p.text,
    options: {
      color: p.color || baseOptions.color,
      bold: p.bold !== undefined ? p.bold : baseOptions.bold,
      fontFace: FONT
    }
  }));
  slide.addText(runs, Object.assign({}, baseOptions, box));
}

function addBydHeader(slide, pres, C, data) {
  const B = bydColors(C);
  slide.background = { color: B.bg };
  slide.addText(data.brand || "BYD", {
    x: 0.18, y: 0.10, w: 0.8, h: 0.22,
    fontSize: 13, fontFace: FONT, color: "707070", bold: true,
    align: "left", valign: "middle", margin: 0
  });
  slide.addText(data.section || "策略页", {
    x: 1.05, y: 0.12, w: 1.5, h: 0.20,
    fontSize: 10, fontFace: FONT, color: "666666", bold: true,
    align: "left", valign: "middle", margin: 0
  });
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0.15, y: 0.35, w: 9.7, h: 0.02,
    fill: { color: B.line }, line: { color: B.line }
  });
}

function addRedBar(slide, pres, C, text, x, y, w, h, opts = {}) {
  const B = bydColors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: opts.fill || B.red }, line: { color: opts.fill || B.red }
  });
  slide.addText(text || "", {
    x: x + 0.06, y: y + 0.02, w: w - 0.12, h: h - 0.04,
    fontSize: opts.fontSize || 11, fontFace: FONT,
    color: opts.color || B.white, bold: true,
    align: opts.align || "center", valign: "middle", margin: 0
  });
}

function addBlackBar(slide, pres, C, text, x, y, w, h, opts = {}) {
  const B = bydColors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: opts.fill || B.black }, line: { color: opts.fill || B.black }
  });
  slide.addText(text || "", {
    x: x + 0.06, y: y + 0.02, w: w - 0.12, h: h - 0.04,
    fontSize: opts.fontSize || 8.5, fontFace: FONT,
    color: opts.color || B.white, bold: true,
    align: opts.align || "center", valign: "middle", margin: 0
  });
}

function addPlainBox(slide, pres, C, x, y, w, h, opts = {}) {
  const B = bydColors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: opts.fill || B.card, transparency: opts.transparency || 0 },
    line: { color: opts.line || B.line, width: opts.lineWidth || 0.5 }
  });
}

function addTextBox(slide, C, text, x, y, w, h, opts = {}) {
  const B = bydColors(C);
  slide.addText(text || "", {
    x, y, w, h,
    fontSize: opts.fontSize || 9, fontFace: FONT,
    color: opts.color || B.text,
    bold: !!opts.bold,
    align: opts.align || "left",
    valign: opts.valign || "middle",
    margin: opts.margin !== undefined ? opts.margin : 0.04,
    fit: opts.fit || "shrink",
    breakLine: opts.breakLine !== undefined ? opts.breakLine : false,
    lineSpacingMultiple: opts.lineSpacingMultiple || 1.1
  });
}

function addPlaceholder(slide, pres, C, label, x, y, w, h) {
  const B = bydColors(C);
  addPlainBox(slide, pres, C, x, y, w, h, { fill: B.lightGray, line: B.line });
  slide.addText(label || "图片占位", {
    x: x + 0.06, y: y + h / 2 - 0.1, w: w - 0.12, h: 0.2,
    fontSize: 7, fontFace: FONT, color: "8A8A8A",
    align: "center", valign: "middle", margin: 0
  });
}

async function addBydImage(slide, pres, C, img, x, y, w, h, label) {
  if (img && typeof img === "string") {
    await addScreenshot(slide, img, x, y, w, h);
  } else if (img && img.path) {
    await addScreenshot(slide, img.path, x, y, w, h);
  } else {
    addPlaceholder(slide, pres, C, label, x, y, w, h);
  }
}

function addArrowBanner(slide, pres, C, text, x, y, w, h) {
  const B = bydColors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w: w - 0.28, h,
    fill: { color: B.red }, line: { color: B.red }
  });
  slide.addShape(pres.shapes.CHEVRON, {
    x: x + w - 0.42, y, w: 0.42, h,
    fill: { color: B.red }, line: { color: B.red }
  });
  slide.addText(text || "", {
    x: x + 0.08, y: y + 0.03, w: w - 0.55, h: h - 0.06,
    fontSize: 10.5, fontFace: FONT, color: B.white, italic: true,
    bold: true, align: "center", valign: "middle", margin: 0
  });
}

function addTableCell(slide, pres, C, text, x, y, w, h, opts = {}) {
  const B = bydColors(C);
  slide.addShape(pres.shapes.RECTANGLE, {
    x, y, w, h,
    fill: { color: opts.fill || B.white },
    line: { color: opts.line || B.line, width: opts.lineWidth || 0.5 }
  });
  if (opts.richText && opts.richText.length) {
    addBydRichText(slide, opts.richText, { x: x + 0.04, y: y + 0.03, w: w - 0.08, h: h - 0.06 }, {
      fontSize: opts.fontSize || 8, fontFace: FONT, color: opts.color || B.text,
      bold: !!opts.bold, align: opts.align || "center", valign: opts.valign || "middle", margin: 0
    });
  } else {
    slide.addText(text || "", {
      x: x + 0.04, y: y + 0.03, w: w - 0.08, h: h - 0.06,
      fontSize: opts.fontSize || 8, fontFace: FONT, color: opts.color || B.text,
      bold: !!opts.bold, align: opts.align || "center", valign: opts.valign || "middle",
      margin: 0.02, fit: opts.fit || "shrink", breakLine: false, lineSpacingMultiple: 1.05
    });
  }
}

function addSocialDots(slide, pres, x, y) {
  const items = [
    { t: "抖", c: "FFD84D" },
    { t: "微", c: "F5C542" },
    { t: "小", c: "7BAAF7" },
    { t: "红", c: "FF5A5F" },
    { t: "头", c: "E53935" }
  ];
  items.forEach((it, idx) => {
    slide.addShape(pres.shapes.ELLIPSE, {
      x: x + idx * 0.28, y, w: 0.22, h: 0.22,
      fill: { color: it.c }, line: { color: it.c }
    });
    slide.addText(it.t, {
      x: x + idx * 0.28, y: y + 0.01, w: 0.22, h: 0.18,
      fontSize: 7, fontFace: FONT, color: "000000", bold: true,
      align: "center", valign: "middle", margin: 0
    });
  });
}

// -----------------------------------------------------
// T45: BYD核心优势三栏洞察页
// -----------------------------------------------------
async function createBydCoreValueInsightSlide(pres, C, data) {
  // data = {
  //   section, brand,
  //   title, titleParts,
  //   columns:[{ title, subTitle, image, emph, note }]
  // }
  const slide = pres.addSlide();
  const B = bydColors(C);
  addBydHeader(slide, pres, C, { brand: data.brand || "BYD", section: data.section || "背景洞察" });

  addPlainBox(slide, pres, C, 0.15, 0.45, 9.7, 0.36, { fill: B.softGray, line: B.softGray });
  if (data.titleParts && data.titleParts.length) {
    addBydRichText(slide, data.titleParts, { x: 0.30, y: 0.51, w: 9.4, h: 0.22 }, {
      fontSize: 10.5, fontFace: FONT, color: B.text, bold: true, align: "left", valign: "middle", margin: 0
    });
  } else {
    slide.addText(data.title || "核心优势素即将加推新款，搭载最新技术、越级配置于一身，将对同价位竞品形成“价值压制”", {
      x: 0.30, y: 0.51, w: 9.4, h: 0.22, fontSize: 10.5, fontFace: FONT,
      color: B.text, bold: true, align: "left", margin: 0
    });
  }

  const cols = data.columns || [];
  const gap = 0.10;
  const x0 = 0.15;
  const totalW = 9.7;
  const cardW = (totalW - gap * 2) / 3;
  for (let i = 0; i < 3; i++) {
    const col = cols[i] || {};
    const x = x0 + i * (cardW + gap);
    addRedBar(slide, pres, C, col.title || `卖点${i + 1}`, x, 0.92, cardW, 0.34, { fontSize: 10 });
    addBlackBar(slide, pres, C, col.subTitle || "小标题说明", x, 1.30, cardW, 0.22, { fontSize: 7.5 });
    await addBydImage(slide, pres, C, col.image, x, 1.56, cardW, 1.75, `配图 ${i + 1}`);
    slide.addText(col.emph || "带来同级“舒适性碾压式”体验", {
      x: x + 0.08, y: 3.37, w: cardW - 0.16, h: 0.24,
      fontSize: 10, fontFace: FONT, color: B.text, bold: true,
      align: "center", valign: "middle", margin: 0
    });
    slide.addText(col.note || "用于补充解释这一个卖点的短说明。", {
      x: x + 0.08, y: 3.66, w: cardW - 0.16, h: 0.48,
      fontSize: 7.3, fontFace: FONT, color: B.mid,
      align: "center", valign: "top", margin: 0.02, fit: "shrink"
    });
  }
}
```

### 使用要点

1. `titleParts` 可做标题中的红色高亮词。
2. 每个 `columns[i]` 支持 `title / subTitle / image / emph / note` 五个字段。
3. 适合将“技术普惠 / 续航破局 / 安全标杆”这类三大卖点做并列呈现。
