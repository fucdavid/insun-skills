 */

  const FONT_FACE = data.fontFace || (typeof FONT !== "undefined" ? FONT : "Microsoft YaHei");

  const shapeLib = pres.shapes || pres.ShapeType || {};
  const RECT = shapeLib.RECTANGLE || shapeLib.rect || "rect";
  const ROUND_RECT = shapeLib.ROUND_RECT || shapeLib.roundRect || "roundRect";
  const LINE = shapeLib.LINE || shapeLib.line || "line";

  const CC = Object.assign({
    bg: "FFFFFF",

    orangeText: "F36B22",
    cyan: "27B8BE",

    border: "777777",
    lightGrey: "F0F0F0",
    panelGrey: "F6F6F6",
    calloutGrey: "737985",
    yellow: "FFF200"
  }, C || {});

  const defaultD = {
    title: "昊铂KOC 5月运营规划",
    headline: "五月KOC将从五一假期、S600 上市、520节日三大重点事项发力，合力凸显昊铂新豪华调性",

    columns: [

        direction: "方向一：挑战测评类      方向二：体验类分享",

            { text: "计划邀约" },

          ],

            { text: "条线索" }
          ]
        ]
      },
      {
        topCaption: "结合新车试驾凸显智慧豪华",
        barTitle: "S600 上 市 营 销 助 力",
        hashtag: "#打CALL新豪华智慧运动SUV",
        desc: "邀请用户到店试驾 S600 分享试驾口碑，并主动邀约首批车主开始分享用车体验，展现智慧新豪华实力",
        direction: "方向一：试驾口碑分享   方向二：首提体验分享",
        images: [
          { label: "试驾口碑图" },
          { label: "首提体验图" }
        ],
        callout: [
          [
            { text: "计划邀请" },
            { text: "xx", highlight: true },
            { text: "名用户，产出试驾口碑" }
          ],
          [
            { text: "XX", highlight: true },
            { text: "篇，产出首提体验" },
            { text: "XX", highlight: true },

          ]

      },
      {
        topCaption: "结合生活方式凸显新豪华格调",
        barTitle: "5月热点节日借势",
        hashtag: "#520昊铂和你一起大声说爱",
        desc: "邀请用户在 520 节点，趣味利用昊铂空间对另一半大声表达爱，如利用后备箱布置花海，利用储物空间送礼物，利用智能座舱当背景音乐等",
        direction: "惊喜后备箱        车内送礼",

          { label: "车内送礼图" }
        ],
        callout: [
          [
            { text: "计划邀请" },
            { text: "xx", highlight: true },
            { text: "名用户，产出试驾口碑" }
          ],
          [
            { text: "XX", highlight: true },
            { text: "篇，产出首提体验" },
            { text: "XX", highlight: true },
            { text: "篇" }
          ]
        ]
      }
    ],
    bottomText: "邀约产出方式强化：任务发布邀约 + 优质创作者1V1邀约 + 公域创作非入库车主主动邀约"
  };

  const D = Object.assign({}, defaultD, data || {});
  D.logos = Object.assign({}, defaultD.logos, (data || {}).logos || {});
  D.columns = (data && data.columns) || defaultD.columns;

  slide.background = { color: CC.bg };

  function addText(txt, x, y, w, h, opt = {}) {
    slide.addText(txt == null ? "" : txt, Object.assign({
      x, y, w, h,
      fontFace: FONT_FACE,
      fontSize: 8,
      color: CC.textDark,
      bold: false,
      align: "left",
      valign: "top",
      margin: 0,
      fit: "shrink"
    }, opt || {}));
  }

  function addRect(x, y, w, h, fill, line) {
    slide.addShape(RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addRoundRect(x, y, w, h, fill, line) {
    slide.addShape(ROUND_RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addLine(x, y, w, h, color = "999999", width = 0.6, lineOpt = {}) {
    slide.addShape(LINE, {
      x, y, w, h,
      line: Object.assign({ color, width }, lineOpt || {})
    });
  }

  function addHyptecAionLogo(logos = {}, x = 7.82, y = 0.30) {
    if (logos.hyptecPath) {
      slide.addImage({ path: logos.hyptecPath, x, y, w: 1.05, h: 0.34 });
    } else {
      addText("◆", x, y + 0.01, 0.24, 0.25, {
        fontSize: 15,
        color: CC.orange,
        bold: true,
        align: "center",
        valign: "middle"
      });

      addText("广汽昊铂\nHYPTEC", x + 0.23, y - 0.01, 0.75, 0.35, {
        fontSize: 7.2,
        color: CC.orange,
        bold: true,
        align: "left",
        valign: "middle",
        lineSpacingMultiple: 0.8
      });
    }

    addLine(x + 1.05, y + 0.02, 0, 0.30, "8F8F8F", 0.8);

    if (logos.aionPath) {
      slide.addImage({ path: logos.aionPath, x: x + 1.18, y: y + 0.01, w: 0.96, h: 0.30 });
    } else {
      addText("AION", x + 1.18, y - 0.01, 1.03, 0.34, {
        fontSize: 15.5,
        color: CC.cyan,
        bold: true,
        align: "left",
        valign: "middle"
      });
    }
  }

  function addImageOrPlaceholder(img, x, y, w, h, fallbackLabel = "图片占位", opt = {}) {
    const path = typeof img === "string" ? img : (img && img.path);
    const label = (img && (img.label || img.caption || img.placeholder)) || fallbackLabel;

    if (path) {
      slide.addImage({ path, x, y, w, h });

      if (img && img.overlayText) {
        addText(img.overlayText, x + 0.05, y + h - 0.35, w - 0.10, 0.30, {
          fontSize: opt.overlayFontSize || 7,
          color: opt.overlayColor || "FFFFFF",
          bold: true,
          align: "center",
          valign: "middle"
        });
      }
    } else {
      addRect(x, y, w, h, opt.fill || "EDEDED", { color: opt.lineColor || "D0D0D0", width: 0.6 });
      addText(label, x + 0.05, y + h / 2 - 0.10, w - 0.10, 0.20, {
        fontSize: opt.fontSize || 6.5,
        color: opt.color || "888888",
        bold: true,
        align: "center",
        valign: "middle"
      });
    }
  }

  function estimateTextWidth(text, fontSize) {
    let units = 0;
    String(text || "").split("").forEach(ch => {
      if (/[\u4e00-\u9fa5]/.test(ch)) units += 1.0;
      else if (ch === " ") units += 0.35;
      else units += 0.55;
    });
    return Math.max(0.02, units * fontSize / 72 * 0.90);
  }

  function addInlineParts(parts, x, y, w, h, opt = {}) {
    const fontSize = opt.fontSize || 8;
    const baseColor = opt.color || CC.textDark;
    const highlightBg = opt.highlightBg || CC.yellow;
    const highlightColor = opt.highlightColor || "FFFFFF";

    if (!Array.isArray(parts)) {
      addText(parts || "", x, y, w, h, {
        fontSize,
        color: baseColor,
        bold: opt.bold,
        align: opt.align || "center",
        valign: "middle"
      });
      return;
    }

    let widths = parts.map(p => estimateTextWidth(p.text || "", fontSize));
    const total = widths.reduce((a, b) => a + b, 0);
    const scale = total > w ? w / total : 1;
    widths = widths.map(v => v * scale);

    let curX = x;
    if ((opt.align || "center") === "center") {
      curX = x + Math.max(0, (w - widths.reduce((a, b) => a + b, 0)) / 2);
    }

    parts.forEach((part, i) => {
      const tw = widths[i];
      const isHighlight = part.highlight || part.bg;
      const bg = part.bg || (isHighlight ? highlightBg : null);

      if (bg) {
        addRect(curX + 0.01, y + 0.02, Math.max(0.05, tw - 0.02), h - 0.03, bg, {
          color: bg,
          transparency: 100
        });
      }

      addText(part.text || "", curX, y, tw, h, {
        fontSize,
        color: isHighlight ? highlightColor : baseColor,
        bold: part.bold != null ? part.bold : opt.bold,
        align: "center",
        valign: "middle"
      });

      curX += tw;
    });
  }

  // 顶部标题与 Logo
  addText(D.title, 0.35, 0.32, 4.30, 0.34, {
    fontSize: 15,
    bold: true,
    color: CC.textDark,
    align: "left",
    valign: "middle"
  });

  addHyptecAionLogo(D.logos, 7.82, 0.30);
  addLine(0.35, 0.63, 9.35, 0, CC.line, 0.6);

  // 中央主标题
  addText(D.headline, 0.80, 0.88, 8.90, 0.42, {
    fontSize: 13.5,
    bold: true,
    color: "000000",
    align: "center",
    valign: "middle"
  });

  function addPlanningColumn(col, x, w) {
    addText(col.topCaption || "", x, 1.28, w, 0.25, {
      fontSize: 10.3,
      bold: true,
      color: CC.textLight,
      align: "center",
      valign: "middle"
    });

    addRoundRect(x, 1.52, w, 0.43, CC.orange);
    addText(col.barTitle || "", x, 1.58, w, 0.30, {
      fontSize: 12,
      bold: true,
      color: "FFFFFF",
      align: "center",
      valign: "middle",
      charSpace: 1.2
    });

    addRect(x, 2.06, w, 2.78, "FFFFFF", { color: CC.border, width: 0.6 });

    addText(col.hashtag || "", x + 0.08, 2.15, w - 0.16, 0.24, {
      fontSize: 10.5,
      bold: true,
      color: "000000",
      align: "center",
      valign: "middle"
    });

    addText(col.desc || "", x + 0.16, 2.39, w - 0.32, 0.52, {
      fontSize: 7.0,
      color: "000000",
      align: "left",
      valign: "top",
      lineSpacingMultiple: 1.08
    });

    addText(col.direction || "", x + 0.12, 2.88, w - 0.24, 0.20, {
      fontSize: 8.4,
      bold: true,
      color: "000000",
      align: "center",
      valign: "middle"
    });

    const imgY = 3.12;
    const imgH = 1.39;
    const imgW = 1.28;
    const imgGap = 0.18;
    const imgX1 = x + 0.24;
    const imgX2 = imgX1 + imgW + imgGap;

    addImageOrPlaceholder((col.images || [])[0], imgX1, imgY, imgW, imgH, "图片1", {
      fill: "E7ECEF",
      lineColor: "D0D0D0"
    });

    addImageOrPlaceholder((col.images || [])[1], imgX2, imgY, imgW, imgH, "图片2", {
      fill: "E7ECEF",
      lineColor: "D0D0D0"
    });

    const calloutX = x + 0.53;
    const calloutY = 4.62;
    const calloutW = w - 1.06;
    const calloutH = 0.43;

    addRoundRect(calloutX, calloutY, calloutW, calloutH, CC.calloutGrey);

    const lines = col.callout || [];

    addInlineParts(lines[0] || "", calloutX + 0.05, calloutY + 0.05, calloutW - 0.10, 0.17, {
      fontSize: 8.6,
      color: "FFFFFF",
      bold: false,
      align: "center",
      highlightBg: CC.yellow,
      highlightColor: "FFFFFF"
    });

    addInlineParts(lines[1] || "", calloutX + 0.05, calloutY + 0.22, calloutW - 0.10, 0.17, {
      fontSize: 8.6,
      color: "FFFFFF",
      bold: false,
      align: "center",
      highlightBg: CC.yellow,
      highlightColor: "FFFFFF"
    });
  }

  const colW = 2.96;
  const xs = [0.35, 3.53, 6.70];

  (D.columns || []).slice(0, 3).forEach((col, i) => {
    addPlanningColumn(col, xs[i], colW);
  });

  addRect(0.35, 5.18, 9.30, 0.33, CC.lightGrey);
  addText(D.bottomText, 0.48, 5.20, 9.04, 0.28, {
    fontSize: 10.4,
    bold: true,
    color: "000000",
    align: "center",
    valign: "middle"
  });

  addLine(0, 5.60, 10, 0, "C8EAEA", 1.0);
}
