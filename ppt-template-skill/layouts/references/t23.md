## T23: 月度活动日历页（12月份环形/双行排列 + 每月活动名称卡片 + 底部城市标签）

适合：全年月度活动规划、经销商月度主题活动展示。按月份排列，每月一个主题活动卡片，底部可标注覆盖城市。

```javascript
function createMonthlyCalendarSlide(pres, C, data) {
  // data = { title, summary, months: [{ month: "3月", activity: "活动名" }], cities: ["北京",...], bottomNote }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.summary) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.85, w: 9, h: 0.4, fill: { color: C.primary } });
    slide.addText(data.summary, {
      x: 0.7, y: 0.85, w: 8.6, h: 0.4,
      fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  // 月度卡片（上行6个 + 下行6个，形成U型时间流）
  const months = data.months || [];
  const row1 = months.slice(0, 6);  // 上行：左→右
  const row2 = months.slice(6, 12); // 下行：右→左（倒序展示）

  const cardGap = 0.15;
  const cardW = (9.0 - cardGap * 5) / 6;
  const row1Y = 1.5;
  const row2Y = 3.3;
  const cardH = 1.4;

  // 上行（左→右）
  row1.forEach((m, i) => {
    const cx = 0.5 + i * (cardW + cardGap);
    // 月份标签
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: row1Y, w: cardW, h: 0.3,
      fill: { color: C.primary }
    });
    slide.addText(m.month, {
      x: cx, y: row1Y, w: cardW, h: 0.3,
      fontSize: 9, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    // 活动名卡片
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: row1Y + 0.3, w: cardW, h: cardH - 0.3,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    slide.addText(m.activity || "", {
      x: cx + 0.05, y: row1Y + 0.35, w: cardW - 0.1, h: cardH - 0.4,
      fontSize: 8, fontFace: FONT, color: C.textDark, align: "center", valign: "middle",
      lineSpacingMultiple: 1.3, margin: 0
    });
    // 连接箭头（非最后一个）
    if (i < row1.length - 1) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx + cardW, y: row1Y + cardH / 2 - 0.01, w: cardGap, h: 0.02,
        fill: { color: C.border }
      });
    }
  });

  // 右侧竖向连接线（上行最后→下行最后）
  if (row2.length > 0) {
    const connX = 0.5 + 5 * (cardW + cardGap) + cardW / 2;
    slide.addShape(pres.shapes.RECTANGLE, {
      x: connX - 0.01, y: row1Y + cardH, w: 0.02, h: row2Y - row1Y - cardH,
      fill: { color: C.border }
    });
  }

  // 下行（右→左）
  row2.forEach((m, i) => {
    const cx = 0.5 + (5 - i) * (cardW + cardGap); // 从右到左
    // 月份标签
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: row2Y, w: cardW, h: 0.3,
      fill: { color: C.primary }
    });
    slide.addText(m.month, {
      x: cx, y: row2Y, w: cardW, h: 0.3,
      fontSize: 9, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    // 活动名卡片
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: row2Y + 0.3, w: cardW, h: cardH - 0.3,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    slide.addText(m.activity || "", {
      x: cx + 0.05, y: row2Y + 0.35, w: cardW - 0.1, h: cardH - 0.4,
      fontSize: 8, fontFace: FONT, color: C.textDark, align: "center", valign: "middle",
      lineSpacingMultiple: 1.3, margin: 0
    });
    // 连接箭头（非最后一个）
    if (i < row2.length - 1) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx - cardGap, y: row2Y + cardH / 2 - 0.01, w: cardGap, h: 0.02,
        fill: { color: C.border }
      });
    }
  });

  // 底部城市标签
  const cities = data.cities || [];
  if (cities.length > 0) {
    const cityY = row2Y + cardH + 0.25;
    const cityGap = 0.1;
    const cityW = Math.min(1.1, (9.0 - cityGap * (cities.length - 1)) / cities.length);
    const cityStartX = 5.0 - (cities.length * (cityW + cityGap) - cityGap) / 2;
    cities.forEach((city, i) => {
      const ccx = cityStartX + i * (cityW + cityGap);
      slide.addShape(pres.shapes.OVAL, {
        x: ccx, y: cityY, w: cityW, h: 0.4,
        fill: { color: C.primary }
      });
      slide.addText(city, {
        x: ccx, y: cityY, w: cityW, h: 0.4,
        fontSize: 8, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
      });
    });
  }
}
```
