## T17: 泳道甘特图/总控MAP页（顶部时间轴 + 多行泳道，每行标注关键活动）

适合：年度运营总控图、项目排期表、多维度并行计划展示

```javascript
function createSwimLaneMapSlide(pres, C, data) {
  // data = { title, months: ["3月","4月",...], lanes: [{ label, items: [{ startCol, endCol, text, highlight? }] }] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.3, y: 0.2, w: 0.06, h: 0.4, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.55, y: 0.2, w: 5, h: 0.4,
    fontSize: 18, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  const months = data.months;
  const nMonths = months.length;
  const labelW = 1.2; // 左侧标签列宽
  const chartX = labelW + 0.3;
  const chartW = 10 - chartX - 0.3;
  const colW = chartW / nMonths;
  const headerY = 0.75;

  // 月份表头
  months.forEach((m, i) => {
    slide.addText(m, {
      x: chartX + i * colW, y: headerY, w: colW, h: 0.3,
      fontSize: 7, fontFace: FONT, color: C.textMid, bold: true, align: "center", valign: "middle", margin: 0
    });
  });

  // 表头下分隔线
  slide.addShape(pres.shapes.RECTANGLE, { x: chartX, y: headerY + 0.3, w: chartW, h: 0.01, fill: { color: C.border } });

  // 泳道
  const lanes = data.lanes;
  const laneStartY = headerY + 0.4;
  const laneH = Math.min(0.9, (5.3 - laneStartY) / lanes.length);

  lanes.forEach((lane, li) => {
    const ly = laneStartY + li * laneH;

    // 泳道标签（左侧深色背景）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 0.3, y: ly + 0.05, w: labelW, h: laneH - 0.1,
      fill: { color: C.dark }
    });
    slide.addText(lane.label, {
      x: 0.3, y: ly + 0.05, w: labelW, h: laneH - 0.1,
      fontSize: 8, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 泳道分隔线
    slide.addShape(pres.shapes.RECTANGLE, {
      x: chartX, y: ly + laneH, w: chartW, h: 0.005, fill: { color: C.border }
    });

    // 活动条
    (lane.items || []).forEach(item => {
      const ix = chartX + item.startCol * colW;
      const iw = (item.endCol - item.startCol + 1) * colW - 0.05;
      const barColor = item.highlight ? C.primary : C.primaryLight;
      const textColor = item.highlight ? "FFFFFF" : C.textDark;

      slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
        x: ix, y: ly + laneH * 0.15, w: iw, h: laneH * 0.7,
        fill: { color: barColor }, rectRadius: 0.08
      });
      slide.addText(item.text, {
        x: ix + 0.05, y: ly + laneH * 0.15, w: iw - 0.1, h: laneH * 0.7,
        fontSize: 6, fontFace: FONT, color: textColor, align: "center", valign: "middle", margin: 0
      });
    });
  });
}
```
