## T7: 详细表格页

```javascript
function createTableSlide(pres, C, data) {
  // data = { title, subtitle, headerRow[], valueRow[], detailTables: [{ title, rows: [[]] }] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };

  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.2, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.2, w: 7, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.headerRow && data.valueRow) {
    const n = data.headerRow.length;
    const hCells = data.headerRow.map(h => ({ text: h, options: { bold: true, fontSize: 9, fontFace: FONT, color: C.textDark, align: "center", valign: "middle" } }));
    const vCells = data.valueRow.map(v => ({ text: v, options: { bold: true, fontSize: 13, fontFace: FONT, color: C.primary, align: "center", valign: "middle" } }));
    slide.addTable([hCells, vCells], {
      x: 0.5, y: 0.85, w: 9, border: { pt: 0.5, color: C.border }, colW: Array(n).fill(9 / n), fill: { color: C.cardBg }
    });
  }

  if (data.detailTables) {
    const dtCount = data.detailTables.length;
    const dtW = 9.0 / dtCount;
    data.detailTables.forEach((dt, i) => {
      const dx = 0.5 + i * dtW;
      slide.addShape(pres.shapes.RECTANGLE, { x: dx + 0.05, y: 1.9, w: dtW - 0.1, h: 0.35, fill: { color: C.dark } });
      slide.addText(dt.title, {
        x: dx + 0.05, y: 1.9, w: dtW - 0.1, h: 0.35,
        fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
      });
      const rows = dt.rows.map(row => row.map(cell => ({ text: cell, options: { fontSize: 8, fontFace: FONT, color: C.textDark, align: "center", valign: "middle" } })));
      slide.addTable(rows, {
        x: dx + 0.05, y: 2.3, w: dtW - 0.1,
        border: { pt: 0.5, color: C.border }, colW: [(dtW - 0.1) * 0.55, (dtW - 0.1) * 0.45], fill: { color: C.cardBg }
      });
    });
  }
}
```
