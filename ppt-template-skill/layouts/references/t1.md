## T1: 封面页

```javascript
function createCoverSlide(pres, C, data) {
  // data = { title, subtitle, author }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  // 顶部主色条
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 右下装饰块
  slide.addShape(pres.shapes.RECTANGLE, { x: 6.5, y: 2.5, w: 4, h: 3.5, fill: { color: C.primary, transparency: 85 } });

  // 主标题
  slide.addText(data.title, {
    x: 0.8, y: 1.4, w: 7, h: 1.0,
    fontSize: 38, fontFace: FONT, color: "FFFFFF", bold: true, align: "left", margin: 0
  });

  // 副标题
  slide.addText(data.subtitle || "", {
    x: 0.8, y: 2.5, w: 5.5, h: 0.6,
    fontSize: 18, fontFace: FONT, color: C.primary, align: "left", margin: 0
  });

  // 细横线
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.8, y: 3.3, w: 1.2, h: 0.03, fill: { color: C.primary } });

  // 作者
  slide.addText(data.author || "", {
    x: 0.8, y: 3.6, w: 5, h: 0.4,
    fontSize: 13, fontFace: FONT, color: C.textLight, align: "left", margin: 0
  });
}
```
