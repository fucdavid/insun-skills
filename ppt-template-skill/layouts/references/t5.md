## T5: 左右对比页

```javascript
function createComparisonSlide(pres, C, data) {
  // data = { title, left: { label, items[] }, right: { label, items[] }, conclusion }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };

  slide.addText(data.title, {
    x: 1, y: 0.3, w: 8, h: 0.6,
    fontSize: 24, fontFace: FONT, color: C.textDark, bold: true, align: "center", margin: 0
  });

  // 左卡片
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0.5, y: 1.2, w: 4.2, h: 3.0,
    fill: { color: C.cardBg }, shadow: makeShadow()
  });
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 1.2, w: 4.2, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.left.label, {
    x: 0.5, y: 1.2, w: 4.2, h: 0.45,
    fontSize: 15, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
  });
  const leftText = data.left.items.map((t, i) => ({
    text: t, options: { fontSize: 12, color: C.textDark, breakLine: i < data.left.items.length - 1 }
  }));
  slide.addText(leftText, {
    x: 0.8, y: 1.85, w: 3.6, h: 2.1, fontFace: FONT, align: "left", lineSpacingMultiple: 1.8, margin: 0
  });

  // 右卡片
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 5.3, y: 1.2, w: 4.2, h: 3.0,
    fill: { color: C.cardBg }, shadow: makeShadow()
  });
  slide.addShape(pres.shapes.RECTANGLE, { x: 5.3, y: 1.2, w: 4.2, h: 0.45, fill: { color: C.textMid } });
  slide.addText(data.right.label, {
    x: 5.3, y: 1.2, w: 4.2, h: 0.45,
    fontSize: 15, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
  });
  const rightText = data.right.items.map((t, i) => ({
    text: t, options: { fontSize: 12, color: C.textMid, breakLine: i < data.right.items.length - 1 }
  }));
  slide.addText(rightText, {
    x: 5.6, y: 1.85, w: 3.6, h: 2.1, fontFace: FONT, align: "left", lineSpacingMultiple: 1.8, margin: 0
  });

  // 底部总结
  if (data.conclusion) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 4.85, w: 10, h: 0.6, fill: { color: C.dark } });
    slide.addText(data.conclusion, {
      x: 0.5, y: 4.85, w: 9, h: 0.6,
      fontSize: 16, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }
}
```
