## T24: 流程推进页（横向3-5个阶段大色块，每阶段下方带详细要点列表+箭头连接）

适合：运营推进流程、营销活动模板推广路径、从试点到规模化的阶段推进。每阶段一个色块，箭头串联。

```javascript
function createProcessFlowSlide(pres, C, data) {
  // data = { title, summary, stages: [{ label, items: ["要点1",...] }] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.summary) {
    slide.addText(data.summary, {
      x: 0.75, y: 0.8, w: 8.5, h: 0.35,
      fontSize: 10, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  }

  const stages = data.stages;
  const n = stages.length;
  const arrowW = 0.3;
  const totalArrows = n - 1;
  const totalW = 9.0;
  const stageW = (totalW - arrowW * totalArrows) / n;
  const stageY = 1.3;
  const labelH = 0.5;
  const contentH = 3.5;

  stages.forEach((stage, i) => {
    const sx = 0.5 + i * (stageW + arrowW);

    // 阶段标签（主色条）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: sx, y: stageY, w: stageW, h: labelH,
      fill: { color: C.primary }
    });
    slide.addText(stage.label, {
      x: sx, y: stageY, w: stageW, h: labelH,
      fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 内容卡片
    slide.addShape(pres.shapes.RECTANGLE, {
      x: sx, y: stageY + labelH, w: stageW, h: contentH,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });

    // 要点列表
    const items = stage.items || [];
    items.forEach((item, j) => {
      const iy = stageY + labelH + 0.15 + j * 0.42;
      // 小色块圆点
      slide.addShape(pres.shapes.OVAL, {
        x: sx + 0.12, y: iy + 0.06, w: 0.12, h: 0.12,
        fill: { color: C.primary }
      });
      slide.addText(item, {
        x: sx + 0.3, y: iy, w: stageW - 0.42, h: 0.35,
        fontSize: 7, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
      });
    });

    // 箭头连接（非最后一个）
    if (i < n - 1) {
      const arrowX = sx + stageW;
      const arrowMidY = stageY + (labelH + contentH) / 2;
      // 箭头横线
      slide.addShape(pres.shapes.RECTANGLE, {
        x: arrowX, y: arrowMidY - 0.01, w: arrowW - 0.05, h: 0.02,
        fill: { color: C.primary }
      });
      // 箭头三角
      slide.addShape(pres.shapes.RECTANGLE, {
        x: arrowX + arrowW - 0.15, y: arrowMidY - 0.08, w: 0.16, h: 0.16,
        fill: { color: C.primary }, rotate: 45
      });
    }
  });
}
```
