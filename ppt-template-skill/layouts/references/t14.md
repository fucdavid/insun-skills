## T14: 策略框架页（顶部主题 + N列策略卡片 + 底部统一基座条）

适合：多维度并行策略展示，如"4大运营方向"、"3大业务支柱"等，底部有统一支撑说明。

```javascript
function createStrategyFrameworkSlide(pres, C, data) {
  // data = { title, subtitle, pillars: [{ tag, name, desc }], foundation, bottomNote }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 顶部主题标题（居中大字，带主色高亮效果）
  slide.addText(data.title, {
    x: 1, y: 0.2, w: 8, h: 0.6,
    fontSize: 24, fontFace: FONT, color: C.textDark, bold: true, align: "center", margin: 0
  });
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 1, y: 0.8, w: 8, h: 0.4,
      fontSize: 12, fontFace: FONT, color: C.textMid, align: "center", margin: 0
    });
  }

  // 策略卡片列
  const pillars = data.pillars;
  const n = pillars.length;
  const gap = 0.2;
  const totalW = 9.0;
  const pillarW = (totalW - gap * (n - 1)) / n;
  const pillarY = 1.4;
  const pillarH = 2.6;

  pillars.forEach((p, i) => {
    const px = 0.5 + i * (pillarW + gap);

    // 卡片背景（深色）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: px, y: pillarY, w: pillarW, h: pillarH,
      fill: { color: C.dark }, shadow: makeShadow()
    });

    // 顶部标签条（主色）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: px, y: pillarY, w: pillarW, h: 0.5,
      fill: { color: C.primary }
    });
    slide.addText(p.tag || "", {
      x: px, y: pillarY, w: pillarW, h: 0.5,
      fontSize: 14, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 策略名称
    slide.addText(p.name, {
      x: px + 0.15, y: pillarY + 0.65, w: pillarW - 0.3, h: 0.4,
      fontSize: 11, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });

    // 描述文字
    slide.addText(p.desc || "", {
      x: px + 0.15, y: pillarY + 1.1, w: pillarW - 0.3, h: pillarH - 1.3,
      fontSize: 8, fontFace: FONT, color: C.textLight, align: "left", lineSpacingMultiple: 1.5, margin: 0
    });
  });

  // 底部统一基座条（红底白字，跨全宽）
  if (data.foundation) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 4.15, w: totalW, h: 0.4, fill: { color: C.primary } });
    slide.addText(data.foundation, {
      x: 0.5, y: 4.15, w: totalW, h: 0.4,
      fontSize: 9, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  // 最底部注释条（深色底）
  if (data.bottomNote) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 4.65, w: totalW, h: 0.35, fill: { color: C.dark } });
    slide.addText(data.bottomNote, {
      x: 0.5, y: 4.65, w: totalW, h: 0.35,
      fontSize: 8, fontFace: FONT, color: C.textLight, align: "center", valign: "middle", margin: 0
    });
  }
}
```
