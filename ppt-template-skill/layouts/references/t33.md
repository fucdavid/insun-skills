/**
 * T33: OTA传播——新媒体领域
 * 适合：OTA传播规划、新媒体传播矩阵、产品升级传播节奏、预热-直播-持续传播三段式规划页
 *
 * data = {
 *   title,
 *   logoText,
 *   logoPath,
 *   topNotes: [
 *     { label, highlight, suffix },
 *     { label, prefix, tech, emotion }
 *   ],
 *   goals: [
 *     { header, title, sub },
 *     { header, title, sub }
 *   ],
 *   themeBar,
 *   sideLabels: { goal, strategy, use },
 *   phases: [
 *     {
 *       phase,
 *       date,
 *       timelineText,
 *       timelineHighlight,
 *       title,
 *       subTitle,
 *       bodyTitle,
 *       body
 *     }
 *   ],
 *   bottomText
 * }
 */
function createT33OtaNewMediaSlide(pres, C = {}, data = {}) {
  const slide = pres.addSlide();

  const FONT_FACE = data.fontFace || (typeof FONT !== "undefined" ? FONT : "Microsoft YaHei");
  const shapeLib = pres.shapes || pres.ShapeType || {};
  const RECT = shapeLib.RECTANGLE || shapeLib.rect || "rect";
  const LINE = shapeLib.LINE || shapeLib.line || "line";
  const RIGHT_ARROW = shapeLib.RIGHT_ARROW || shapeLib.rightArrow || "rightArrow";

  const CC = Object.assign({
    bg: "FFFFFF",
    primary: "009CA2",
    primaryDark: "00999C",
    primaryLight: "DDF6F4",
    primaryPale: "EFFBFA",
    textDark: "111111",
    textMid: "333333",
    textLight: "666666",
    border: "20B9B5",
    grey: "AFAFAF",
    lightGrey: "F4F4F4",
    accent: "FFC000",
    logo: "168B97",
    line: "9E9E9E"
  }, C || {});

  const D = Object.assign({
    title: "OTA传播——新媒体领域",
    logoText: "东风奕派",
    logoPath: "",
    topNotes: [
      {
        label: "【三元矩阵联动】",
        highlight: "品牌官号+领导人IP+直播矩阵",
        suffix: "三管齐下；"
      },
      {
        label: "【双线场景释放】",
        prefix: "",
        tech: "硬核技术线-NOA产品力释放；",
        emotion: "感性情感线-NOA对车主实际生活需求的纾解与治愈。"
      }
    ],
    goals: [
      {
        header: "维度一：产品曝光种草",
        title: "007+ 年轻人的第一台高智性能轿跑",
        sub: "©微博：XXX"
      },
      {
        header: "维度二：品牌形象提升",
        title: "体现奕派品牌在智能化领域的再一次跃升",
        sub: "©讨论：XXXX"
      }
    ],
    themeBar: "#奕路进阶，智享新境——奕派 007+ OTA 升级全场景体验",
    sideLabels: {
      goal: "目标",
      strategy: "达成策略",
      use: "使用"
    },
    phases: [
      {
        phase: "预热启动",
        date: "5.1",
        timelineText: "领导预热视频：奕派OTA焕新升级",
        timelineHighlight: "",
        title: "造期待",
        subTitle: "以领导视频预热视频\n建立用户对OTA升级的“技术好奇”与“场景想象”",
        bodyTitle: "高管下场，全方位清晰释疑",
        body: "仅围绕调书 + 用户真实共鸣两个角度设计爆点话题与预热视频内容，为本次迭代做权威背书，扩大影响力，展现品牌自信与实力。"
      },
      {
        phase: "直播引爆",
        date: "5.8",
        timelineText: "直播事件：",
        timelineHighlight: "奕派007+ NOA轻生活挑战日",
        title: "秀体验",
        subTitle: "打造“武汉NOA轻生活一日游”直播\n真实路线沉浸体验，引发用户对NOA升级即时认同",
        bodyTitle: "情感线：",
        body: "迎合年轻人生活痛点，展现NOA功能带来的「轻松体验、掌控感、安全感」，自动泊入复杂车位，一小时之内，集合城市通勤和休闲生活全场景。\n\n技术线：\n①城市/高速NOA实测：左靠左道超车巡航——城市NOA（自动启停/跟车、识别路况、人车混行、右转占道）+ 高速NOA（自动变道、车流汇入）\n②轻松生活场景组合：视频场景+探店闲逛+车内休憩，完成“技术-生活”的情绪承接。"
      },
      {
        phase: "持续传播",
        date: "5.10",
        timelineText: "产品力视频：OTA全维升级",
        timelineHighlight: "",
        title: "夯认知",
        subTitle: "视频小TVC视频深度解析\n专业垂直讲解，夯实用户对OTA升级背后的价值认知",
        bodyTitle: "模拟TVC二创混剪，借助人设强化认知",
        body: "解析产品素材进行再包装，生成深度TVC内容，以及专业垂直 + 情绪讲解，实现更受关注OTA升级的技术细节。"
      }
    ],
    bottomText: "视频*4支（15万） + 直播*1场（30万） = 总价（45万）"
  }, data || {});

  slide.background = { color: CC.bg };

  function addRect(x, y, w, h, fill, line) {
    slide.addShape(RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addLine(x, y, w, h, color = CC.line, width = 0.6, dash) {
    slide.addShape(LINE, {
      x, y, w, h,
      line: {
        color,
        width,
        dash
      }
    });
  }

  function addText(txt, x, y, w, h, opt = {}) {
    slide.addText(txt == null ? "" : txt, Object.assign({
      x, y, w, h,
      fontFace: FONT_FACE,
      fontSize: 8,
      color: CC.textDark,
      margin: 0,
      bold: false,
      align: "left",
      valign: "top",
      fit: "shrink"
    }, opt || {}));
  }

  function addVerticalLabel(text, x, y, w, h, fontSize = 8.5) {
    addRect(x, y, w, h, CC.primaryDark);
    addText(Array.from(text || "").join("\n"), x, y + 0.06, w, h - 0.12, {
      fontSize,
      bold: true,
      color: "FFFFFF",
      align: "center",
      valign: "middle",
      breakLine: false,
      lineSpacingMultiple: 0.75
    });
  }

  function addGoalBox(goal, x, y, w) {
    addRect(x, y, w, 0.27, CC.primaryLight);
    addText(goal.header || "", x, y + 0.04, w, 0.18, {
      fontSize: 7.4,
      color: CC.primary,
      bold: true,
      align: "center",
      valign: "middle"
    });

    addRect(x, y + 0.29, w, 0.48, "FFFFFF", { color: "D5F0EF", width: 0.5 });
    addText(goal.title || "", x + 0.05, y + 0.37, w - 0.10, 0.17, {
      fontSize: 7.2,
      bold: true,
      color: CC.textDark,
      align: "center",
      valign: "middle"
    });
    addText(goal.sub || "", x + 0.05, y + 0.55, w - 0.10, 0.12, {
      fontSize: 5.3,
      color: CC.textLight,
      align: "center",
      valign: "middle"
    });
  }

  function addArrowBar(text, x, y, w, h) {
    slide.addShape(RIGHT_ARROW, {
      x, y, w, h,
      fill: { color: CC.grey },
      line: { color: CC.grey, transparency: 100 }
    });
    addText(text || "", x, y + 0.03, w - 0.05, h - 0.05, {
      fontSize: 8.3,
      color: "FFFFFF",
      bold: true,
      align: "center",
      valign: "middle"
    });
  }

  function addTimeline(phase, x, y, w) {
    addLine(x + 0.05, y + 0.12, w - 0.12, 0, "B6B6B6", 0.7);
    addText("●", x + 0.07, y + 0.02, 0.15, 0.15, {
      fontSize: 8.2,
      color: CC.primary,
      align: "center",
      valign: "middle"
    });
    addText(phase.date || "", x + 0.02, y - 0.03, 0.30, 0.14, {
      fontSize: 5.8,
      color: CC.textDark,
      align: "center",
      valign: "middle"
    });

    if (phase.timelineHighlight) {
      addText([
        { text: phase.timelineText || "", options: { color: CC.textDark, bold: true } },
        { text: phase.timelineHighlight || "", options: { color: CC.primary, bold: true } }
      ], x + 0.42, y - 0.01, w - 0.42, 0.22, {
        fontSize: 6.3,
        align: "left",
        valign: "middle"
      });
    } else {
      addText(phase.timelineText || "", x + 0.42, y - 0.01, w - 0.42, 0.22, {
        fontSize: 6.3,
        bold: true,
        color: CC.textDark,
        align: "left",
        valign: "middle"
      });
    }
  }

  function addPhaseColumn(phase, x, y, w, h) {
    // 标题浅青底
    addRect(x, y, w, 0.83, CC.primaryPale);
    addText(phase.title || "", x, y + 0.00, w, 0.38, {
      fontSize: 20,
      bold: true,
      italic: true,
      color: CC.primary,
      align: "center",
      valign: "middle"
    });
    addRect(x + w / 2 - 0.36, y + 0.40, 0.72, 0.025, CC.primary);

    addText(phase.subTitle || "", x + 0.08, y + 0.47, w - 0.16, 0.30, {
      fontSize: 6.5,
      bold: true,
      color: CC.textDark,
      align: "center",
      valign: "middle",
      lineSpacingMultiple: 1.0
    });

    // 内容框
    addRect(x, y + 0.83, w, h - 0.83, "FFFFFF", { color: CC.border, width: 0.8 });
    addText("■  " + (phase.bodyTitle || ""), x + 0.15, y + 0.98, w - 0.30, 0.22, {
      fontSize: 6.5,
      bold: true,
      color: CC.textDark,
      align: "left",
      valign: "top"
    });
    addText(phase.body || "", x + 0.15, y + 1.38, w - 0.30, h - 1.50, {
      fontSize: 5.75,
      color: CC.textDark,
      align: "left",
      valign: "top",
      breakLine: false,
      lineSpacingMultiple: 1.05
    });
  }

  // 顶部标题
  addText(D.title, 0.37, 0.18, 5.0, 0.26, {
    fontSize: 12.8,
    bold: true,
    color: CC.textDark,
    align: "left",
    valign: "middle"
  });

  // 右上角 Logo
  if (D.logoPath) {
    slide.addImage({
      path: D.logoPath,
      x: 8.45,
      y: 0.15,
      w: 1.25,
      h: 0.30
    });
  } else {
    addText("◆", 8.50, 0.13, 0.28, 0.30, {
      fontSize: 15,
      color: CC.logo,
      bold: true,
      align: "center",
      valign: "middle"
    });
    addText(D.logoText || "", 8.78, 0.16, 0.95, 0.26, {
      fontSize: 10.2,
      bold: true,
      color: CC.logo,
      align: "left",
      valign: "middle"
    });
  }

  // 顶部细线
  addLine(0.37, 0.57, 9.30, 0, "888888", 0.6);

  // 顶部两条策略说明
  const n0 = D.topNotes[0] || {};
  addText([
    { text: "□  ", options: { bold: true, color: "000000" } },
    { text: n0.label || "", options: { bold: true, color: "000000" } },
    { text: n0.highlight || "", options: { bold: true, color: CC.accent } },
    { text: n0.suffix || "", options: { bold: true, color: "000000" } }
  ], 0.43, 0.72, 9.10, 0.16, {
    fontSize: 7.5,
    align: "left",
    valign: "middle"
  });

  const n1 = D.topNotes[1] || {};
  addText([
    { text: "□  ", options: { bold: true, color: "000000" } },
    { text: n1.label || "", options: { bold: true, color: "000000" } },
    { text: n1.prefix || "", options: { bold: true, color: "000000" } },
    { text: n1.tech || "", options: { bold: true, color: "000000" } },
    { text: n1.emotion || "", options: { bold: true, color: "000000" } }
  ], 0.43, 0.90, 9.15, 0.16, {
    fontSize: 7.5,
    align: "left",
    valign: "middle"
  });

  // 左侧目标标签
  addVerticalLabel(D.sideLabels.goal || "目标", 0.38, 1.16, 0.40, 0.90, 8.5);

  // 双维度目标框
  addGoalBox((D.goals || [])[0] || {}, 0.95, 1.16, 3.95);
  addGoalBox((D.goals || [])[1] || {}, 5.22, 1.16, 4.45);

  // 左侧达成策略标签
  addVerticalLabel(D.sideLabels.strategy || "达成策略", 0.38, 2.08, 0.40, 3.05, 8.2);

  // 主题条
  addRect(0.95, 2.08, 8.72, 0.34, CC.primaryDark);
  addText(D.themeBar || "", 0.95, 2.11, 8.72, 0.26, {
    fontSize: 8.2,
    color: "FFFFFF",
    bold: true,
    align: "center",
    valign: "middle"
  });

  // 三阶段箭头
  const phases = D.phases || [];
  const colY = 3.08;
  const colH = 2.05;
  const colW = 2.86;
  const colGap = 0.16;
  const x1 = 0.95;
  const x2 = x1 + colW + colGap;
  const x3 = x2 + colW + colGap;

  addArrowBar((phases[0] || {}).phase || "", x1, 2.50, colW, 0.31);
  addArrowBar((phases[1] || {}).phase || "", x2, 2.50, colW, 0.31);
  addArrowBar((phases[2] || {}).phase || "", x3, 2.50, colW, 0.31);

  // 时间线
  addTimeline(phases[0] || {}, x1, 2.83, colW);
  addTimeline(phases[1] || {}, x2, 2.83, colW);
  addTimeline(phases[2] || {}, x3, 2.83, colW);

  // 分隔虚线
  addLine(x2 - 0.08, 2.90, 0, 2.20, CC.border, 0.7, "dash");
  addLine(x3 - 0.08, 2.90, 0, 2.20, CC.border, 0.7, "dash");

  // 三列内容
  addPhaseColumn(phases[0] || {}, x1, colY, colW, colH);
  addPhaseColumn(phases[1] || {}, x2, colY, colW, colH);
  addPhaseColumn(phases[2] || {}, x3, colY, colW, colH);

  // 左下使用标签
  addRect(0.38, 5.14, 0.40, 0.28, CC.primaryDark);
  addText(D.sideLabels.use || "使用", 0.38, 5.14, 0.40, 0.28, {
    fontSize: 6.5,
    bold: true,
    color: "FFFFFF",
    align: "center",
    valign: "middle"
  });

  // 底部预算 / 使用说明
  addRect(0.95, 5.14, 8.72, 0.28, "FFFFFF", { color: CC.border, width: 0.8 });
  addText(D.bottomText || "", 0.95, 5.16, 8.72, 0.22, {
    fontSize: 7.2,
    bold: true,
    color: CC.textDark,
    align: "center",
    valign: "middle"
  });

  // 页脚浅青收边
  addRect(0, 5.56, 10, 0.06, "DFF7F6");
}
