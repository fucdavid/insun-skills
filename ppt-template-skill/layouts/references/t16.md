## T16: 多品牌对标总结页（N列品牌并排，每列品牌名+关键词+大号标签）

适合：竞品分析总结、行业调研小结，并排对比多个品牌的核心运营策略

```javascript
function createBrandComparisonSlide(pres, C, data) {
  // data = { title, subtitle, brands: [{ name, keyword, bigLabel }] }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  // 顶部总结文字
  if (data.title) {
    slide.addText(data.title, {
      x: 0.5, y: 0.4, w: 9, h: 0.5,
      fontSize: 14, fontFace: FONT, color: C.textLight, align: "center", margin: 0
    });
  }
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 0.5, y: 0.9, w: 9, h: 0.6,
      fontSize: 18, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", margin: 0
    });
  }

  // 品牌列
  const brands = data.brands;
  const n = brands.length;
  const gap = 0.4;
  const totalW = 9.0;
  const colW = (totalW - gap * (n - 1)) / n;

  brands.forEach((b, i) => {
    const bx = 0.5 + i * (colW + gap);
    const by = 2.0;

    // 品牌名（白色，居中）
    slide.addText(b.name, {
      x: bx, y: by, w: colW, h: 0.5,
      fontSize: 14, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 关键词描述
    slide.addText(b.keyword || "", {
      x: bx, y: by + 0.55, w: colW, h: 0.4,
      fontSize: 10, fontFace: FONT, color: C.textLight, align: "center", valign: "middle", margin: 0
    });

    // 大号标签（主色高亮）
    slide.addText(b.bigLabel, {
      x: bx, y: by + 1.1, w: colW, h: 0.8,
      fontSize: 28, fontFace: FONT, color: C.primary, bold: true, align: "center", valign: "middle", margin: 0
    });

    // 分隔竖线（非最后一个）
    if (i < n - 1) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: bx + colW + gap / 2, y: by, w: 0.01, h: 2.0, fill: { color: C.textLight, transparency: 60 }
      });
    }
  });
}
```
