## T19: 现状诊断页（已完成项 ✓ + 待完善项虚线框，状态检查风格）

适合：项目现状分析、能力成熟度评估、已有成果vs待建设项对比展示。上方展示已达成状态（打勾标签），下方展示待完善事项（虚线框）。

```javascript
function createStatusCheckSlide(pres, C, data) {
  // data = { title, intro, doneItems: [{ label, desc }], todoItems: [{ label, desc }], bottomNote }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 引导说明
  if (data.intro) {
    slide.addText(data.intro, {
      x: 0.75, y: 0.8, w: 8.5, h: 0.35,
      fontSize: 11, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  }

  // 已完成项（主色圆角标签 + 白字）
  const doneItems = data.doneItems || [];
  const doneN = doneItems.length;
  const doneGap = 0.25;
  const doneW = (9.0 - doneGap * (doneN - 1)) / doneN;
  const doneY = 1.35;

  doneItems.forEach((item, i) => {
    const dx = 0.5 + i * (doneW + doneGap);
    // 圆角标签背景
    slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
      x: dx, y: doneY, w: doneW, h: 0.5,
      fill: { color: C.primary }, rectRadius: 0.1
    });
    // ✓ + 文字
    slide.addText("✓  " + item.label, {
      x: dx + 0.1, y: doneY, w: doneW - 0.2, h: 0.5,
      fontSize: 12, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    // 描述
    if (item.desc) {
      slide.addText(item.desc, {
        x: dx + 0.1, y: doneY + 0.55, w: doneW - 0.2, h: 0.3,
        fontSize: 8, fontFace: FONT, color: C.textMid, align: "center", valign: "middle", margin: 0
      });
    }
  });

  // 中间总结条
  const midY = doneY + 1.05;
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: midY, w: 9.0, h: 0.45, fill: { color: C.primaryLight } });
  slide.addText(data.bottomNote || "", {
    x: 0.7, y: midY, w: 8.6, h: 0.45,
    fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 待完善项（虚线边框卡片 + 深色背景）
  const todoItems = data.todoItems || [];
  const todoN = todoItems.length;
  const todoGap = 0.25;
  const todoW = (9.0 - todoGap * (todoN - 1)) / todoN;
  const todoY = midY + 0.7;
  const todoH = 5.2 - todoY;

  todoItems.forEach((item, i) => {
    const tx = 0.5 + i * (todoW + todoGap);
    // 深色卡片背景（用实线边框模拟虚线效果）
    slide.addShape(pres.shapes.RECTANGLE, {
      x: tx, y: todoY, w: todoW, h: todoH,
      fill: { color: C.dark },
      line: { color: C.textLight, width: 1.5, dashType: "dash" }
    });
    // 标题
    slide.addText(item.label, {
      x: tx + 0.15, y: todoY + 0.2, w: todoW - 0.3, h: 0.5,
      fontSize: 14, fontFace: FONT, color: C.primary, bold: true, align: "center", valign: "middle", margin: 0
    });
    // 描述
    if (item.desc) {
      slide.addText(item.desc, {
        x: tx + 0.15, y: todoY + 0.75, w: todoW - 0.3, h: todoH - 1.0,
        fontSize: 9, fontFace: FONT, color: C.textLight, align: "center", valign: "middle",
        lineSpacingMultiple: 1.5, margin: 0
      });
    }
  });
}
```
