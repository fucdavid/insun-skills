## T22: 区域组织架构页（左侧地图/区域示意 + 右侧层级流程图）

适合：全国区域布局、经销商网络体系、组织架构与层级关系展示。左侧用色块示意区域分布，右侧用流程图展示层级关系。

```javascript
function createRegionalOrgSlide(pres, C, data) {
  // data = { title, summary, regions: [{ name }], orgFlow: [{ level, label, desc }], bottomNote }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.summary) {
    slide.addText(data.summary, {
      x: 0.75, y: 0.8, w: 8.5, h: 0.35,
      fontSize: 10, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  }

  // 左侧：区域分布示意（用圆形色块网格代替地图）
  const regions = data.regions || [];
  const regStartX = 0.5, regStartY = 1.4;
  const regAreaW = 4.0, regAreaH = 3.8;

  // 背景区域
  slide.addShape(pres.shapes.RECTANGLE, {
    x: regStartX, y: regStartY, w: regAreaW, h: regAreaH,
    fill: { color: C.cardBg }, shadow: makeShadow()
  });

  // 区域圆点分布
  const rCols = 3, rRows = Math.ceil(regions.length / rCols);
  const rGapX = regAreaW / (rCols + 1), rGapY = regAreaH / (rRows + 1);
  regions.forEach((reg, i) => {
    const col = i % rCols, row = Math.floor(i / rCols);
    const rx = regStartX + rGapX * (col + 1);
    const ry = regStartY + rGapY * (row + 1);
    slide.addShape(pres.shapes.OVAL, {
      x: rx - 0.35, y: ry - 0.35, w: 0.7, h: 0.7,
      fill: { color: C.primary, transparency: 20 }
    });
    slide.addText(reg.name, {
      x: rx - 0.5, y: ry - 0.15, w: 1.0, h: 0.3,
      fontSize: 8, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
  });

  // 右侧：组织层级流程图（从上到下的箭头流）
  const orgFlow = data.orgFlow || [];
  const orgX = 5.2, orgW = 4.3;
  const orgStartY = 1.4;
  const orgTotalH = 3.8;
  const levelH = orgTotalH / orgFlow.length;

  orgFlow.forEach((level, i) => {
    const ly = orgStartY + i * levelH;
    // 层级卡片
    slide.addShape(pres.shapes.RECTANGLE, {
      x: orgX, y: ly + 0.05, w: orgW, h: levelH * 0.55,
      fill: { color: i === 0 ? C.primary : C.dark }
    });
    slide.addText(level.label, {
      x: orgX + 0.15, y: ly + 0.05, w: orgW - 0.3, h: levelH * 0.55,
      fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    // 描述
    if (level.desc) {
      slide.addText(level.desc, {
        x: orgX + 0.1, y: ly + levelH * 0.55 + 0.08, w: orgW - 0.2, h: levelH * 0.35,
        fontSize: 7, fontFace: FONT, color: C.textMid, align: "center", valign: "top", margin: 0
      });
    }
    // 向下箭头（非最后一个）
    if (i < orgFlow.length - 1) {
      const arrowY = ly + levelH - 0.12;
      slide.addShape(pres.shapes.RECTANGLE, {
        x: orgX + orgW / 2 - 0.015, y: arrowY, w: 0.03, h: 0.15,
        fill: { color: C.primary }
      });
      // 箭头三角
      slide.addShape(pres.shapes.RECTANGLE, {
        x: orgX + orgW / 2 - 0.08, y: arrowY + 0.12, w: 0.16, h: 0.16,
        fill: { color: C.primary }, rotate: 45
      });
    }
  });

  // 底部注释
  if (data.bottomNote) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 5.1, w: 9.0, h: 0.35, fill: { color: C.primaryLight } });
    slide.addText(data.bottomNote, {
      x: 0.7, y: 5.1, w: 8.6, h: 0.35,
      fontSize: 9, fontFace: FONT, color: C.textDark, align: "center", valign: "middle", margin: 0
    });
  }
}
```
