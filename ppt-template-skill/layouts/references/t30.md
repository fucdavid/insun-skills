## T30: 策略宣言+双目标页（上方引导文 + 中间超大Slogan + 下方左右两个目标卡片）

适合：核心策略发布、运营理念阐述、品牌Slogan展示。上方有引导文字说明背景，中间突出超大的策略关键词/Slogan，下方左右分列短期目标和长期目标（或两个并列的策略方向）。

```javascript
function createStrategySloganSlide(pres, C, data) {
  // data = {
  //   title,
  //   leadText,        // 上方引导语
  //   slogan,          // 中间超大Slogan文字
  //   sloganSub,       // Slogan下方补充说明
  //   leftGoal: { label, title, desc, keywords },   // 左侧目标卡片
  //   rightGoal: { label, title, desc, keywords },   // 右侧目标卡片
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title || "", {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 引导语
  if (data.leadText) {
    slide.addText(data.leadText, {
      x: 1, y: 0.85, w: 8, h: 0.4,
      fontSize: 11, fontFace: FONT, color: C.textMid, align: "center", valign: "middle", margin: 0
    });
  }

  // 超大Slogan
  slide.addText(data.slogan || "", {
    x: 0.5, y: 1.4, w: 9, h: 1.2,
    fontSize: 42, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
  });

  // Slogan下方补充
  if (data.sloganSub) {
    slide.addText(data.sloganSub, {
      x: 1, y: 2.6, w: 8, h: 0.35,
      fontSize: 10, fontFace: FONT, color: C.textMid, align: "center", valign: "middle", margin: 0
    });
  }

  // 左右目标卡片
  const cardY = 3.2;
  const cardW = 4.2;
  const cardH = 2.0;
  const goals = [
    { pos: 0.5, data: data.leftGoal, color: C.primary },
    { pos: 5.3, data: data.rightGoal, color: C.dark }
  ];

  goals.forEach((g) => {
    if (!g.data) return;
    const gd = g.data;

    // 卡片背景
    slide.addShape(pres.shapes.RECTANGLE, {
      x: g.pos, y: cardY, w: cardW, h: cardH,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    // 顶部色条
    slide.addShape(pres.shapes.RECTANGLE, {
      x: g.pos, y: cardY, w: cardW, h: 0.05,
      fill: { color: g.color }
    });

    // 标签（如"短期目标"）
    if (gd.label) {
      slide.addText(gd.label, {
        x: g.pos + 0.2, y: cardY + 0.15, w: cardW - 0.4, h: 0.3,
        fontSize: 12, fontFace: FONT, color: g.color, bold: true, align: "left", valign: "middle", margin: 0
      });
    }

    // 标题（如"让用户来得勤"）
    if (gd.title) {
      slide.addText(gd.title, {
        x: g.pos + 0.2, y: cardY + 0.5, w: cardW - 0.4, h: 0.3,
        fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
      });
    }

    // 描述
    if (gd.desc) {
      slide.addText(gd.desc, {
        x: g.pos + 0.2, y: cardY + 0.85, w: cardW - 0.4, h: 0.5,
        fontSize: 9, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
        lineSpacingMultiple: 1.4, margin: 0
      });
    }

    // 关键词（底部强调）
    if (gd.keywords) {
      slide.addText(gd.keywords, {
        x: g.pos + 0.2, y: cardY + cardH - 0.45, w: cardW - 0.4, h: 0.3,
        fontSize: 10, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
      });
    }
  });
}
```
