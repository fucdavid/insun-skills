          { date: "", topic: "进行中", label: "传播进行中", caption: "4.16-5.24 多话题持续传播" }
        ]
      }

    testDrive: {
      title: "③昊铂S600试驾会传播：",
      summary: "3条内容投放400元，总互动量1107",

        { label: "试驾内容1", caption: "@128车评\n投放1条\n总互动量508" },
        { label: "试驾内容2", caption: "@雅西姐上米卡\n未投放\n总互动量11" },
        { label: "试驾内容3", caption: "@明昊铂的Dr.钱\n投放1条\n总互动量585" }
      ]
    },
    challenge: "存在课题：事项多话题多，但产出不够聚焦，未形成沉淀性话题标签，且部分话题用户参与性不高"
  };

  const D = Object.assign({}, defaultD, data || {});
  D.logos = Object.assign({}, defaultD.logos, (data || {}).logos || {});
  D.coreMetrics = (data && data.coreMetrics) || defaultD.coreMetrics;
  D.levelRows = (data && data.levelRows) || defaultD.levelRows;
  D.screenshots = (data && data.screenshots) || defaultD.screenshots;
  D.investMetrics = (data && data.investMetrics) || defaultD.investMetrics;
  D.investTable = (data && data.investTable) || defaultD.investTable;
  D.sections = (data && data.sections) || defaultD.sections;
  D.testDrive = Object.assign({}, defaultD.testDrive, (data || {}).testDrive || {});

  slide.background = { color: CC.bg };

  function addText(txt, x, y, w, h, opt = {}) {
    slide.addText(txt == null ? "" : txt, Object.assign({
      x, y, w, h,
      fontFace: FONT_FACE,
      fontSize: 8,
      color: CC.textDark,
      bold: false,
      align: "left",
      valign: "top",
      margin: 0,
      fit: "shrink"
    }, opt || {}));
  }

  function addRect(x, y, w, h, fill, line) {
    slide.addShape(RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addRoundRect(x, y, w, h, fill, line) {
    slide.addShape(ROUND_RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addLine(x, y, w, h, color = "999999", width = 0.6, lineOpt = {}) {
    slide.addShape(LINE, {
      x, y, w, h,
      line: Object.assign({ color, width }, lineOpt || {})
    });
  }

  function addDot(x, y, size = 0.08, color = CC.cyan) {
    slide.addShape(OVAL, {
      x, y, w: size, h: size,
      fill: { color },
      line: { color, transparency: 100 }
    });
  }

  function addHyptecAionLogo(logos = {}, x = 7.82, y = 0.30) {
    if (logos.hyptecPath) {
      slide.addImage({ path: logos.hyptecPath, x, y, w: 1.05, h: 0.34 });
    } else {
      addText("◆", x, y + 0.01, 0.24, 0.25, {
        fontSize: 15,
        color: CC.orange,
        bold: true,
        align: "center",
        valign: "middle"
      });

      addText("广汽昊铂\nHYPTEC", x + 0.23, y - 0.01, 0.75, 0.35, {
        fontSize: 7.2,
        color: CC.orange,
        bold: true,
        align: "left",
        valign: "middle",
        lineSpacingMultiple: 0.8
      });
    }

    addLine(x + 1.05, y + 0.02, 0, 0.30, "8F8F8F", 0.8);

    if (logos.aionPath) {
      slide.addImage({ path: logos.aionPath, x: x + 1.18, y: y + 0.01, w: 0.96, h: 0.30 });
    } else {
      addText("AION", x + 1.18, y - 0.01, 1.03, 0.34, {
        fontSize: 15.5,
        color: CC.cyan,
        bold: true,
        align: "left",
        valign: "middle"
      });
    }
  }

  function addImageOrPlaceholder(img, x, y, w, h, fallbackLabel = "图片占位", opt = {}) {
    const path = typeof img === "string" ? img : (img && img.path);
    const label = (img && (img.label || img.caption || img.placeholder)) || fallbackLabel;

    if (path) {
      slide.addImage({ path, x, y, w, h });

      if (img && img.overlayText) {
        addText(img.overlayText, x + 0.05, y + h - 0.35, w - 0.10, 0.30, {
          fontSize: opt.overlayFontSize || 7,
          color: opt.overlayColor || "FFFFFF",
          bold: true,
          align: "center",
          valign: "middle"
        });
      }
    } else {
      addRect(x, y, w, h, opt.fill || "EDEDED", { color: opt.lineColor || "D0D0D0", width: 0.6 });
      addText(label, x + 0.05, y + h / 2 - 0.10, w - 0.10, 0.20, {
        fontSize: opt.fontSize || 6.5,
        color: opt.color || "888888",
        bold: true,
        align: "center",
        valign: "middle"
      });
    }
  }

  function addTable(headers, rows, x, y, w, h, colRatios, opt = {}) {
    const lineColor = opt.lineColor || "8A8A8A";
    const headerH = opt.headerH || 0.22;
    const bodyH = (h - headerH) / Math.max(1, rows.length);
    const ratios = colRatios || headers.map(() => 1);
    const sum = ratios.reduce((a, b) => a + b, 0);
    const colWs = ratios.map(r => w * r / sum);

    if (opt.headerFill) {
      addRect(x, y, w, headerH, opt.headerFill, { color: opt.headerFill, transparency: 100 });
    }

    addRect(x, y, w, h, opt.fill || "FFFFFF", { color: lineColor, width: 0.5 });

    let curX = x;
    headers.forEach((head, i) => {
      addText(head, curX + 0.02, y + 0.04, colWs[i] - 0.04, headerH - 0.06, {
        fontSize: opt.headerFontSize || 6.2,
        bold: true,
        align: "center",
        valign: "middle"
      });

      curX += colWs[i];
      if (i < headers.length - 1) addLine(curX, y, 0, h, lineColor, 0.5);
    });

    addLine(x, y + headerH, w, 0, lineColor, 0.5);

    rows.forEach((row, r) => {
      const rowY = y + headerH + r * bodyH;
      curX = x;

      row.forEach((cell, c) => {
        addText(cell, curX + 0.02, rowY + 0.03, colWs[c] - 0.04, bodyH - 0.04, {
          fontSize: opt.bodyFontSize || 6,
          color: opt.bodyColor || CC.textDark,
          align: "center",
          valign: "middle"
        });

        curX += colWs[c];
      });

      if (r < rows.length - 1) {
        addLine(x, rowY + bodyH, w, 0, lineColor, 0.5);
      }
    });
  }

  // 顶部标题与 Logo
  addText(D.title, 0.35, 0.33, 4.40, 0.34, {
    fontSize: 14.2,
    bold: true,
    color: CC.textDark,
    align: "left",
    valign: "middle"
  });

  addHyptecAionLogo(D.logos, 7.82, 0.30);
  addLine(0.35, 0.63, 9.35, 0, CC.line, 0.6);

  // 主标题
  addText(D.headline, 0.64, 0.74, 9.05, 0.32, {
    fontSize: 12.8,
    bold: true,
    color: "000000",
    align: "center",
    valign: "middle"
  });

  // 左右分栏标题
  addText("➢", 0.31, 1.08, 0.18, 0.20, {
    fontSize: 10,
    bold: true,
    align: "center",
    valign: "middle"
  });

    fontSize: 10.2,
    bold: true,
    color: "000000",
    align: "left",
    valign: "middle"
  });

    fontSize: 10,
    bold: true,
    align: "center",
    valign: "middle"
  });

    fontSize: 10.2,
    bold: true,
    color: "000000",
    align: "left",
    valign: "middle"
  });

  function addMetricBox(metric, x, y, w, h) {
    addRoundRect(x, y, w, h, CC.metricBg);

    addText(metric.value || "", x, y + 0.08, w, 0.20, {
      fontSize: 11,
      bold: true,
      color: CC.orangeText,
      align: "center",
      valign: "middle"
    });

      color: CC.textMid,
      align: "center",
      valign: "middle"

      fontSize: 7.0,

      valign: "middle"
    });

    addText(row.count || "", 1.55, ry, 0.40, 0.16, {
      fontSize: 7.6,

      color: i === 0 ? CC.orangeText : "003B7A",
      align: "center",
      valign: "middle"
    });
  });

  addImageOrPlaceholder(D.screenshots[0], 2.42, 2.04, 0.84, 1.43, "截图1", {
    fill: "E8E8E8"
  });

  addImageOrPlaceholder(D.screenshots[1], 3.47, 2.04, 0.84, 1.43, "截图2", {
    fill: "E8E8E8"
  });

  // 投放数据
  const investY = 3.76;
  const investW = 1.32;
  const investGap = 0.12;

  D.investMetrics.slice(0, 4).forEach((m, i) => {
    addMetricBox(m, 0.31 + i * (investW + investGap), investY, investW, 0.48);
  });

  addTable(
    ["投放账号", "投放条数", "投放效果"],
    D.investTable || [],
    0.31,
    4.34,
    4.36,
    0.73,
    [1.3, 1.3, 1.35],
    {
      headerH: 0.20,
      headerFontSize: 6.2,
      bodyFontSize: 5.8,
      lineColor: "777777"
    }
  );

  function addTimelineSection(section, x, y, w) {
    addText(section.title || "", x, y, 3.05, 0.25, {
      fontSize: 10.5,
      color: "000000",
      align: "left",
      valign: "middle"
    });

    if (section.summary) {
      addRect(x + 3.10, y - 0.03, w - 3.10, 0.40, "FFFFFF", {
        color: CC.red,
        width: 0.7
      });

      addText(section.summary, x + 3.18, y + 0.01, w - 3.24, 0.32, {
        fontSize: 6.6,
        color: "000000",
        align: "left",
        valign: "middle",
        lineSpacingMultiple: 0.9
      });
    }

    const timeline = section.timeline || [];
    const lineY = y + 0.58;

    addLine(x + 0.12, lineY, w - 0.20, 0, "9A9A9A", 0.7, {
      endArrowType: "triangle"
    });

    const n = Math.max(1, timeline.length);
    const usableW = w - 1.35;
    const startX = x + 0.75;
    const gap = n > 1 ? usableW / (n - 1) : 0;

    timeline.forEach((item, i) => {
      const px = startX + i * gap;

      addText(item.date || "", px - 0.33, y + 0.33, 0.70, 0.13, {
        fontSize: 6.0,
        color: "000000",
        bold: true,
        align: "center",
        valign: "middle"
      });

      addText(item.topic || "", px - 0.48, y + 0.46, 0.96, 0.20, {
        fontSize: 5.5,
        color: "000000",
        align: "center",
        valign: "middle",
        lineSpacingMultiple: 0.85
      });

      addDot(px - 0.04, lineY - 0.04, 0.08, CC.blueDot);

      addImageOrPlaceholder(item, px - 0.33, y + 0.72, 0.52, 0.42, item.label || "图", {
        fill: "EDEDED",
        fontSize: 5.2
      });

      addText(item.caption || "", px + 0.20, y + 0.72, 0.67, 0.45, {
        fontSize: 5.4,
        color: "000000",
        align: "left",
        valign: "middle",
        lineSpacingMultiple: 0.9
      });
    });
  }

  addTimelineSection(D.sections[0] || {}, 4.93, 1.25, 4.72);
  addTimelineSection(D.sections[1] || {}, 4.93, 2.72, 4.72);

  // S600 试驾传播
  const testDrive = D.testDrive || {};

  addText(testDrive.title || "", 4.93, 4.12, 2.30, 0.25, {
    fontSize: 10.5,
    color: "000000",
    align: "left",
    valign: "middle"
  });

  addRect(7.08, 4.12, 2.57, 0.25, "FFFFFF", {
    color: CC.red,
    width: 0.7
  });

  addText(testDrive.summary || "", 7.15, 4.14, 2.45, 0.19, {
    fontSize: 6.4,
    color: "000000",
    align: "left",
    valign: "middle"
  });

  const tdItems = testDrive.items || [];
  const tdX = [5.25, 6.75, 8.28];

  tdItems.slice(0, 3).forEach((item, i) => {
    addImageOrPlaceholder(item, tdX[i], 4.45, 0.50, 0.62, item.label || "试驾图", {
      fill: "EDEDED",
      fontSize: 5
    });

    addText(item.caption || "", tdX[i] + 0.58, 4.55, 0.82, 0.45, {
      fontSize: 5.4,
      color: "000000",
      align: "left",
      valign: "middle",
      lineSpacingMultiple: 0.9
    });
  });

  addText(D.challenge || "", 0.80, 5.30, 8.85, 0.34, {
    fontSize: 12.2,
    bold: true,
    color: CC.orangeText,
    align: "center",
    valign: "middle"
  });

  addLine(0, 5.60, 10, 0, "C8EAEA", 1.0);
}
