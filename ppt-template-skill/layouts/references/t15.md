## T15: 品牌案例分析页（上半：运营矩阵流程图 / 下半：营销亮点+活动示例）

适合：竞品/行业案例拆解、品牌运营体系展示。上半部分展示架构/流程，下半部分列举亮点。

```javascript
function createCaseStudySlide(pres, C, data) {
  // data = { title, brandName, tagline, keyPoints: [{ label, desc }], highlights: [{ title, desc }], bottomNote }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 5, h: 0.45,
    fontSize: 18, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 品牌标语（主色大字）
  slide.addText(data.tagline || "", {
    x: 0.75, y: 0.8, w: 8, h: 0.4,
    fontSize: 14, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 上半区：关键要点卡片（左右分栏）
  const kps = data.keyPoints || [];
  const kpStartY = 1.4;
  const kpW = 4.2;
  kps.forEach((kp, i) => {
    const kx = i % 2 === 0 ? 0.5 : 5.3;
    const ky = kpStartY + Math.floor(i / 2) * 0.65;
    // 标签
    slide.addShape(pres.shapes.RECTANGLE, {
      x: kx, y: ky, w: 1.4, h: 0.3, fill: { color: C.primary }
    });
    slide.addText(kp.label, {
      x: kx, y: ky, w: 1.4, h: 0.3,
      fontSize: 9, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    // 描述
    slide.addText(kp.desc, {
      x: kx + 1.5, y: ky, w: kpW - 1.5, h: 0.3,
      fontSize: 9, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  });

  // 分隔线
  const sepY = kpStartY + Math.ceil(kps.length / 2) * 0.65 + 0.15;
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: sepY, w: 9, h: 0.01, fill: { color: C.border } });

  // 下半区：亮点展示
  const hlStartY = sepY + 0.2;
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0.5, y: hlStartY, w: 1.6, h: 0.35, fill: { color: C.dark }
  });
  slide.addText("运营亮点", {
    x: 0.5, y: hlStartY, w: 1.6, h: 0.35,
    fontSize: 10, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
  });

  const hls = data.highlights || [];
  const hlGap = 0.2;
  const hlW = (9.0 - hlGap * (hls.length - 1)) / hls.length;
  hls.forEach((hl, i) => {
    const hx = 0.5 + i * (hlW + hlGap);
    const hy = hlStartY + 0.5;
    slide.addShape(pres.shapes.RECTANGLE, {
      x: hx, y: hy, w: hlW, h: 1.2,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    slide.addText(hl.title, {
      x: hx + 0.1, y: hy + 0.08, w: hlW - 0.2, h: 0.3,
      fontSize: 10, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });
    slide.addText(hl.desc || "", {
      x: hx + 0.1, y: hy + 0.4, w: hlW - 0.2, h: 0.7,
      fontSize: 8, fontFace: FONT, color: C.textMid, align: "left", lineSpacingMultiple: 1.4, margin: 0
    });
  });
}
```
