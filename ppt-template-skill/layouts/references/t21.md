## T21: 金字塔分级体系页（三角金字塔 + 每级对应权益和义务）

适合：用户分级管理、会员等级体系、组织层级展示。左侧金字塔分层，右侧每层对应权益/描述。

```javascript
function createPyramidTierSlide(pres, C, data) {
  // data = { title, summary, tiers: [{ name, desc, benefits: ["..."], duties: ["..."] }] }
  // tiers[0] = 顶层（最高级别），tiers[last] = 底层（最低级别）
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  if (data.summary) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.85, w: 9, h: 0.4, fill: { color: C.primaryLight } });
    slide.addText(data.summary, {
      x: 0.7, y: 0.85, w: 8.6, h: 0.4,
      fontSize: 10, fontFace: FONT, color: C.textMid, align: "left", valign: "middle", margin: 0
    });
  }

  const tiers = data.tiers;
  const n = tiers.length;
  const pyStartY = 1.5;
  const pyH = 3.8;
  const tierH = pyH / n;
  const pyLeftX = 0.5;
  const pyW = 3.2; // 金字塔区域宽度

  // 金字塔层级（从上到下，每层越来越宽）
  tiers.forEach((tier, i) => {
    const ty = pyStartY + i * tierH;
    const indent = (n - 1 - i) * (pyW / (2 * n)); // 越高层缩进越大
    const layerW = pyW - 2 * indent;
    const lx = pyLeftX + indent;

    // 层级色块（深→浅渐变用不同透明度模拟）
    const fillColor = i === 0 ? C.primary : (i === 1 ? C.textMid : C.primaryLight);
    const textColor = i <= 1 ? "FFFFFF" : C.textDark;
    slide.addShape(pres.shapes.RECTANGLE, {
      x: lx, y: ty + 0.03, w: layerW, h: tierH - 0.06,
      fill: { color: fillColor }
    });
    slide.addText(tier.name, {
      x: lx, y: ty + 0.03, w: layerW, h: tierH - 0.06,
      fontSize: 10, fontFace: FONT, color: textColor, bold: true, align: "center", valign: "middle", margin: 0
    });
  });

  // 右侧权益/义务详情卡片
  const detailX = 4.0;
  const detailW = 5.6;

  tiers.forEach((tier, i) => {
    const ty = pyStartY + i * tierH;
    // 卡片背景
    slide.addShape(pres.shapes.RECTANGLE, {
      x: detailX, y: ty + 0.03, w: detailW, h: tierH - 0.06,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    // 层级名称
    slide.addText(tier.name, {
      x: detailX + 0.15, y: ty + 0.08, w: 1.2, h: 0.28,
      fontSize: 9, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });
    // 权益（左半）
    const benefitText = (tier.benefits || []).join("、");
    slide.addText("权益: " + benefitText, {
      x: detailX + 0.15, y: ty + 0.38, w: detailW / 2 - 0.2, h: tierH - 0.48,
      fontSize: 7, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
      lineSpacingMultiple: 1.3, margin: 0
    });
    // 义务（右半）
    const dutyText = (tier.duties || []).join("、");
    slide.addText("义务: " + dutyText, {
      x: detailX + detailW / 2 + 0.05, y: ty + 0.38, w: detailW / 2 - 0.2, h: tierH - 0.48,
      fontSize: 7, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
      lineSpacingMultiple: 1.3, margin: 0
    });
  });
}
```
