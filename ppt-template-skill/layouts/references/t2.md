## T2: 目录页

```javascript
function createTocSlide(pres, C, data) {
  // data = { items: ["章节1", ...], activeIndex: 0 }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 5.35, w: 10, h: 0.05, fill: { color: C.primary } });

  // 右下角装饰大字 CONTENTS（半透明，不遮挡内容）
  slide.addText("CONTENTS", {
    x: 4.8, y: 3.8, w: 5, h: 1.5,
    fontSize: 48, fontFace: FONT, color: C.textLight, bold: true,
    align: "right", valign: "bottom", margin: 0, transparency: 80
  });

  const startY = 0.7;
  const lineH = 0.52;
  data.items.forEach((item, i) => {
    const isActive = (i === data.activeIndex);
    // 左竖线
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 2.5, y: startY + i * lineH, w: 0.05, h: 0.32,
      fill: { color: isActive ? C.primary : C.textLight }
    });
    // 文字
    slide.addText(item, {
      x: 2.8, y: startY + i * lineH, w: 5, h: 0.38,
      fontSize: isActive ? 17 : 15, fontFace: FONT,
      color: isActive ? C.primary : "FFFFFF",
      bold: isActive, align: "left", valign: "middle", margin: 0
    });
  });
}
```
