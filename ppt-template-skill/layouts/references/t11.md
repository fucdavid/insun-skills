## T11: 时间轴节点页（上升曲线/折线 + 圆形日期节点 + 事件说明卡片）

适合：产品发展历程、项目里程碑回顾、运营大事记。节点沿曲线排列，交替上下分布。

```javascript
function createMilestoneTimelineSlide(pres, C, data) {
  // data = { title, summary, milestones: [{ date, event, desc }], bottomText }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  const ms = data.milestones;
  const n = ms.length;
  const lineY = 2.9; // 中轴线Y
  const startX = 0.8;
  const endX = 9.2;
  const spanX = endX - startX;

  // 主横线（上升斜线用矩形近似为水平线，实际PPT中更清晰）
  slide.addShape(pres.shapes.RECTANGLE, {
    x: startX, y: lineY, w: spanX, h: 0.03, fill: { color: C.primary }
  });

  ms.forEach((m, i) => {
    const cx = startX + (spanX / (n - 1 || 1)) * i;
    const isAbove = i % 2 === 0; // 交替上下

    // 圆形节点
    slide.addShape(pres.shapes.OVAL, {
      x: cx - 0.22, y: lineY - 0.22, w: 0.44, h: 0.44, fill: { color: C.primary }
    });
    slide.addText(m.date, {
      x: cx - 0.35, y: lineY - 0.22, w: 0.7, h: 0.44,
      fontSize: 8, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });

    // 竖连接线
    const connH = 0.5;
    if (isAbove) {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx - 0.01, y: lineY - 0.22 - connH, w: 0.02, h: connH, fill: { color: C.primary, transparency: 50 }
      });
    } else {
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx - 0.01, y: lineY + 0.22, w: 0.02, h: connH, fill: { color: C.primary, transparency: 50 }
      });
    }

    // 事件卡片
    const cardW = Math.min(spanX / n + 0.1, 1.6);
    const cardH = 1.0;
    let cardY;
    if (isAbove) {
      cardY = lineY - 0.22 - connH - cardH;
    } else {
      cardY = lineY + 0.22 + connH;
    }

    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx - cardW / 2, y: cardY, w: cardW, h: cardH,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });
    // 卡片顶部色条
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx - cardW / 2, y: cardY, w: cardW, h: 0.04, fill: { color: C.primary }
    });
    // 事件标题
    slide.addText(m.event, {
      x: cx - cardW / 2 + 0.08, y: cardY + 0.1, w: cardW - 0.16, h: 0.3,
      fontSize: 9, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });
    // 事件描述
    if (m.desc) {
      slide.addText(m.desc, {
        x: cx - cardW / 2 + 0.08, y: cardY + 0.4, w: cardW - 0.16, h: cardH - 0.5,
        fontSize: 7, fontFace: FONT, color: C.textMid, align: "left", lineSpacingMultiple: 1.3, margin: 0
      });
    }
  });

  // 底部总结条
  if (data.bottomText) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 4.85, w: 10, h: 0.6, fill: { color: C.primary } });
    slide.addText(data.bottomText, {
      x: 0.5, y: 4.85, w: 9, h: 0.6,
      fontSize: 11, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
  }
}
```
