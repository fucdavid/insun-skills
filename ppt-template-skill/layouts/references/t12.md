## T12: 中心辐射生态图页（中心圆 + 环绕子节点 + 外围三大色块标签）

适合：生态体系、组织架构、平台矩阵等中心向外辐射的结构展示

```javascript
function createEcosystemSlide(pres, C, data) {
  // data = { title, centerLabel, innerNodes: ["节点1",...], outerLabels: [{ label, desc }] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题
  slide.addText(data.title, {
    x: 1, y: 0.2, w: 8, h: 0.6,
    fontSize: 22, fontFace: FONT, color: C.textDark, bold: true, align: "center", margin: 0
  });

  // 中心大圆
  const ccx = 5.0, ccy = 3.0, cr = 0.65;
  slide.addShape(pres.shapes.OVAL, {
    x: ccx - cr, y: ccy - cr, w: cr * 2, h: cr * 2,
    fill: { color: C.primary }, shadow: makeShadow()
  });
  slide.addText(data.centerLabel, {
    x: ccx - cr, y: ccy - cr, w: cr * 2, h: cr * 2,
    fontSize: 11, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
  });

  // 环绕内部节点（圆形排列）
  const nodes = data.innerNodes || [];
  const nr = 1.6; // 节点到中心的距离
  const nodeR = 0.35;
  nodes.forEach((nd, i) => {
    const angle = (Math.PI * 2 * i) / nodes.length - Math.PI / 2;
    const nx = ccx + nr * Math.cos(angle);
    const ny = ccy + nr * Math.sin(angle);

    // 连接线（从中心到节点）
    // 用细矩形近似，从中心指向节点
    const dx = nx - ccx, dy = ny - ccy;
    const dist = Math.sqrt(dx * dx + dy * dy);
    const lineLen = dist - cr - nodeR;
    const mx = ccx + (cr / dist) * dx;
    const my = ccy + (cr / dist) * dy;
    // 水平/竖直近似连接线
    slide.addShape(pres.shapes.RECTANGLE, {
      x: Math.min(mx, nx), y: Math.min(my, ny),
      w: Math.max(Math.abs(nx - mx), 0.02), h: Math.max(Math.abs(ny - my), 0.02),
      fill: { color: C.border }
    });

    // 节点圆
    slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
      x: nx - 0.7, y: ny - 0.2, w: 1.4, h: 0.4,
      fill: { color: C.dark }, rectRadius: 0.2
    });
    slide.addText(nd, {
      x: nx - 0.7, y: ny - 0.2, w: 1.4, h: 0.4,
      fontSize: 8, fontFace: FONT, color: "FFFFFF", align: "center", valign: "middle", margin: 0
    });
  });

  // 外围三大标签（左、右上、右下）
  const outerLabels = data.outerLabels || [];
  const positions = [
    { x: 0.3, y: 1.5 },  // 左
    { x: 7.5, y: 1.3 },  // 右上
    { x: 7.5, y: 3.8 },  // 右下
  ];
  outerLabels.forEach((ol, i) => {
    if (i >= 3) return;
    const pos = positions[i];
    slide.addShape(pres.shapes.OVAL, {
      x: pos.x, y: pos.y, w: 1.8, h: 1.0,
      fill: { color: C.primary }, shadow: makeShadow()
    });
    slide.addText([
      { text: ol.label, options: { fontSize: 12, bold: true, breakLine: true } },
      { text: ol.desc || "", options: { fontSize: 8 } }
    ], {
      x: pos.x, y: pos.y, w: 1.8, h: 1.0,
      fontFace: FONT, color: "FFFFFF", align: "center", valign: "middle", margin: 0
    });
  });
}
```
