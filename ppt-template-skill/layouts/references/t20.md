## T20: 双区并列定位页（左右两个区域各自有名称/理念/Slogan，中间用图形连接）

适合：主品牌+子品牌定位展示、双产品线对比定位、官方俱乐部+专属社区并列介绍。中间用嵌套圆/连接图形体现从属关系。

```javascript
function createDualPositionSlide(pres, C, data) {
  // data = { title, subtitle, left: { name, concept, slogan }, right: { name, concept, slogan }, bottomChannels: ["渠道1",...] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 副标题
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 0.75, y: 0.8, w: 8.5, h: 0.35,
      fontSize: 11, fontFace: FONT, color: C.textMid, align: "center", valign: "middle", margin: 0
    });
  }

  // 中央连接图（大圆+小圆嵌套/重叠）
  const centerX = 5.0, centerY = 2.7;
  // 大圆（左侧区域）
  slide.addShape(pres.shapes.OVAL, {
    x: centerX - 1.6, y: centerY - 0.9, w: 1.8, h: 1.8,
    fill: { color: C.primaryLight, transparency: 30 },
    line: { color: C.primary, width: 1.5 }
  });
  slide.addText(data.left.name || "", {
    x: centerX - 1.6, y: centerY - 0.3, w: 1.8, h: 0.6,
    fontSize: 9, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
  });
  // 小圆（右侧子区域，部分重叠）
  slide.addShape(pres.shapes.OVAL, {
    x: centerX - 0.2, y: centerY - 0.65, w: 1.3, h: 1.3,
    fill: { color: C.primary, transparency: 20 },
    line: { color: C.primary, width: 1.5 }
  });
  slide.addText(data.right.name || "", {
    x: centerX - 0.2, y: centerY - 0.15, w: 1.3, h: 0.5,
    fontSize: 8, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
  });

  // 左侧详情区
  const leftX = 0.5, detailY = 1.3;
  slide.addText(data.left.name, {
    x: leftX, y: detailY, w: 3.5, h: 0.4,
    fontSize: 16, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
  });
  slide.addText(data.left.concept || "", {
    x: leftX, y: detailY + 0.45, w: 3.5, h: 0.8,
    fontSize: 9, fontFace: FONT, color: C.textMid, align: "left", lineSpacingMultiple: 1.5, margin: 0
  });
  if (data.left.slogan) {
    slide.addShape(pres.shapes.RECTANGLE, { x: leftX, y: detailY + 1.3, w: 3.2, h: 0.35, fill: { color: C.primaryLight } });
    slide.addText("Slogan: " + data.left.slogan, {
      x: leftX + 0.1, y: detailY + 1.3, w: 3.0, h: 0.35,
      fontSize: 9, fontFace: FONT, color: C.textDark, align: "left", valign: "middle", margin: 0
    });
  }

  // 右侧详情区
  const rightX = 6.5;
  slide.addText(data.right.name, {
    x: rightX, y: detailY, w: 3.5, h: 0.4,
    fontSize: 16, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
  });
  slide.addText(data.right.concept || "", {
    x: rightX, y: detailY + 0.45, w: 3.5, h: 0.8,
    fontSize: 9, fontFace: FONT, color: C.textMid, align: "left", lineSpacingMultiple: 1.5, margin: 0
  });
  if (data.right.slogan) {
    slide.addShape(pres.shapes.RECTANGLE, { x: rightX, y: detailY + 1.3, w: 3.2, h: 0.35, fill: { color: C.primaryLight } });
    slide.addText("Slogan: " + data.right.slogan, {
      x: rightX + 0.1, y: detailY + 1.3, w: 3.0, h: 0.35,
      fontSize: 9, fontFace: FONT, color: C.textDark, align: "left", valign: "middle", margin: 0
    });
  }

  // 底部承载渠道条
  const channels = data.bottomChannels || [];
  if (channels.length > 0) {
    const chY = 4.6;
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.3, y: chY, w: 1.0, h: 0.4, fill: { color: C.dark } });
    slide.addText("承载渠道", {
      x: 0.3, y: chY, w: 1.0, h: 0.4,
      fontSize: 8, fontFace: FONT, color: "FFFFFF", bold: true, align: "center", valign: "middle", margin: 0
    });
    const chGap = 0.15;
    const chW = (8.2 - chGap * (channels.length - 1)) / channels.length;
    channels.forEach((ch, i) => {
      const cx = 1.5 + i * (chW + chGap);
      slide.addShape(pres.shapes.RECTANGLE, {
        x: cx, y: chY, w: chW, h: 0.4,
        fill: { color: C.cardBg }, line: { color: C.border, width: 0.8 }
      });
      slide.addText(ch, {
        x: cx, y: chY, w: chW, h: 0.4,
        fontSize: 8, fontFace: FONT, color: C.textDark, align: "center", valign: "middle", margin: 0
      });
    });
  }
}
```
