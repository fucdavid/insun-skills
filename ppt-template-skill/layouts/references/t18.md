## T18: 三级活动分类日历页（ABC三类活动分层展示 + 时间轴 + 活动矩阵）

适合：年度活动规划总览、分级活动管理，按A类/B类/C类分层展示不同规模活动

```javascript
function createActivityCalendarSlide(pres, C, data) {
  // data = { title, tiers: [{ level, levelDesc, specs, events: [{ month, name }] }] }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.3, y: 0.2, w: 0.06, h: 0.4, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.55, y: 0.2, w: 6, h: 0.4,
    fontSize: 18, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  const tiers = data.tiers;
  const n = tiers.length;
  const tierStartY = 0.8;
  const tierH = (5.0 - tierStartY) / n;

  tiers.forEach((tier, ti) => {
    const ty = tierStartY + ti * tierH;
    const colors = [C.primary, C.textMid, C.border]; // 不同层级不同颜色

    // 左侧层级标签
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 0.3, y: ty + 0.05, w: 1.6, h: tierH - 0.1,
      fill: { color: colors[ti] || C.primary }
    });
    slide.addText([
      { text: tier.level, options: { fontSize: 14, bold: true, breakLine: true } },
      { text: tier.levelDesc || "", options: { fontSize: 7, breakLine: true } },
      { text: tier.specs || "", options: { fontSize: 6 } }
    ], {
      x: 0.3, y: ty + 0.05, w: 1.6, h: tierH - 0.1,
      fontFace: FONT, color: "FFFFFF", align: "center", valign: "middle", margin: [2, 4, 2, 4]
    });

    // 活动卡片网格
    const events = tier.events || [];
    const cardStartX = 2.1;
    const cardAreaW = 7.5;
    const maxPerRow = 4;
    const cardGap = 0.12;
    const cardW = (cardAreaW - cardGap * (maxPerRow - 1)) / maxPerRow;
    const rowH = (tierH - 0.15) / Math.max(Math.ceil(events.length / maxPerRow), 1);

    events.forEach((ev, ei) => {
      const col = ei % maxPerRow;
      const row = Math.floor(ei / maxPerRow);
      const ex = cardStartX + col * (cardW + cardGap);
      const ey = ty + 0.08 + row * rowH;

      slide.addShape(pres.shapes.RECTANGLE, {
        x: ex, y: ey, w: cardW, h: Math.min(rowH - 0.06, 0.6),
        fill: { color: C.cardBg }, shadow: makeShadow()
      });
      slide.addText(ev.name, {
        x: ex + 0.05, y: ey, w: cardW - 0.1, h: Math.min(rowH - 0.06, 0.6),
        fontSize: 7, fontFace: FONT, color: C.textDark, align: "center", valign: "middle", margin: 0
      });
    });
  });
}
```
