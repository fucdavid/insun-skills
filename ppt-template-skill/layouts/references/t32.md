 * 适合：赛事直播规划、活动直播规划、双节点直播方案、直播线+视频线排期页
 *
 * data = {
 *   title,
 *   logoText,
 *   logoPath,
 *   themeTitle,
 *   objective,
 *   strategy,
 *   liveLabel,
 *   cards: [
 *     { dateTitle, barTitle, timeText, subTitle, items: [{ title, desc }] },
 *     { dateTitle, barTitle, timeText, subTitle, items: [{ title, desc }] }
 *   ],
 *   welfare,
 *   videoLabel,
 *   videos: [ "左侧视频线内容", "右侧视频线内容" ]
 * }
 */
function createWuhanMarathonLivePlanSlide(pres, C = {}, data = {}) {
  const slide = pres.addSlide();

  const FONT_FACE = data.fontFace || (typeof FONT !== "undefined" ? FONT : "Microsoft YaHei");
  const shapeLib = pres.shapes || {};
  const RECT = shapeLib.RECTANGLE || "rect";

  const CC = Object.assign({
    bg: "FFFFFF",
    primary: "009CA2",
    primaryDark: "00979C",
    primaryLight: "A9E8E3",
    accent: "E7AF5F",
    textDark: "111111",
    textMid: "333333",
    border: "009CA2",
    rowBg: "F2F2F2",
    sideBg: "EFEFEF",
    videoBg: "F4F4F4",
    cardBg: "FFFFFF",
    logo: "168B97",
    lineDark: "111111"
  }, C || {});

  const D = Object.assign({
    title: "2026年武汉马拉松直播规划方案——整体直播规划",
    logoText: "东风奕派",
    logoPath: "",
    themeTitle: "群樱荟萃，奕起出发——跟着汉马跑进春天",
    objective: "借势马拉松热度，通过“体育精神+产品性能+用户体验”的结合，软性植入奕派007+ / 奕派007 / 奕派008 / 纳米01等车核心卖点。",
    strategy: "做“赛道外的狂欢”，把奕派直播间变成跑者的“第二打卡点”和观众的“赛事啦啦窗”，过程中巧妙植入品牌及车型优势点",
    liveLabel: "直播线",
    cards: [
      {
        dateTitle: "3.19  汉马博览会",
        barTitle: "——聚人气——",
        timeText: "单场2h，上午下午各1场",
        subTitle: "利用马博会开幕式人流量，以“逛展”吸引观看",
        items: [
          {
            title: "展台开场-氛围介绍",
            desc: "展示介绍展台设计理念及车型概况"
          },
          {
            title: "物资发放-跑友采访",
            desc: "展示现场人山人海的领物盛况，采访排队领物的跑者，聊他们明天的目标。"
          },
          {
            title: "游戏挑战-卖点植入",
            desc: "结合跑友必备的物资打包技能，设置008“后备箱补给包打包赛”"
          },
          {
            title: "沉浸逛展-自家展车&友商串门儿",
            desc: "镜头聚焦奕派展车区的“高颜值布置”。展示eπ007等展车，还可前往室内展会逛展互动"
          }
        ]
      },
      {
        dateTitle: "3.22  汉马正式开跑日",
        barTitle: "——爆高潮——",
        timeText: "上午一场（2~3h待定）",
        subTitle: "“庆功直播间”不仅讲车，更要迎接完赛英雄",
        items: [
          {
            title: "终点展台开场-氛围介绍",
            desc: "展示现场奕派元素，及奕派赞助汉马背后的意义"
          },
          {
            title: "鸣枪时刻-央媒推流（待定）",
            desc: "主播观看开跑，重点突出007+领跑车"
          },
          {
            title: "展台讲车-（不时串场）",
            desc: "奕派终点展区，讲车型亮点；可穿插车内健身运动展示，融入体育精神"
          },
          {
            title: "半马展台互动&选手采访",
            desc: "主播讲解008开启"
          },
          {
            title: "终点接待-冠军时刻直播）&领先选手采访",
            desc: "邀请完赛跑者上车休息，打开电动魔毯座椅休憩并采访"
          }
        ]
      }
    ],
    welfare: "福利互动：每20分钟一次倒计时口令福袋 “东风奕派，马力全开” （抖币+汉马周边+奕派商城周边）",
    videoLabel: "视频线",
    videos: [
      "①预热视频《奕派速度，领跑者宣言》\n主播和领跑车合拍，预告直播日\n\n②精彩集锦《奕派带你云逛展》\n记录直播切片亮点瞬间-物资篇",
      "①全程快剪《奕派的汉马高光时刻》\n从鸣枪领跑，半马站点、到全程保障，终点展台等全流程剪辑\n②选手采访集锦《汉马终点的那些故事》\n记录选手们逛奕派展台和在车内接受采访的画面"
    ]
  }, data || {});

  slide.background = { color: CC.bg };

  function addRect(x, y, w, h, fill, line) {
    slide.addShape(RECT, {
      x, y, w, h,
      fill: { color: fill },
      line: line || { color: fill, transparency: 100 }
    });
  }

  function addText(txt, x, y, w, h, opt = {}) {
    const options = Object.assign({
      x, y, w, h,
      fontFace: FONT_FACE,
      fontSize: 8,
      color: CC.textDark,
      bold: false,
      align: "left",
      valign: "top",
      margin: 0
    }, opt || {});

    slide.addText(txt == null ? "" : txt, options);
  }

  function richItems(items) {
    if (!Array.isArray(items)) return items || "";

    const runs = [];
    items.forEach((it, idx) => {
      const title = it.title || "";
      const desc = Array.isArray(it.desc) ? it.desc.join("\n") : (it.desc || "");

      runs.push({
        text: `${idx + 1}.${title}\n`,
        options: { bold: true }
      });

      if (desc) {
        runs.push({
          text: desc + (idx === items.length - 1 ? "" : "\n"),
          options: { bold: false }
        });
      }
    });

    return runs;
  }

  function addVerticalLabel(label, x, y, w, h) {
    addRect(x, y, w, h, CC.sideBg);
    const verticalText = Array.from(label || "").join("\n");

    addText(verticalText, x, y + h / 2 - 0.42, w, 0.84, {
      fontSize: 9,
      bold: true,
      align: "center",
      valign: "middle",
      lineSpacingMultiple: 0.8
    });
  }

  function addInfoRow(label, body, y) {
    addRect(0.38, y, 0.42, 0.27, CC.rowBg);
    addText(label, 0.38, y, 0.42, 0.27, {
      fontSize: 8.5,
      bold: true,
      align: "center",
      valign: "middle"
    });

    addRect(0.86, y, 8.85, 0.27, CC.rowBg);
    addText(body, 0.92, y, 8.72, 0.27, {
      fontSize: 8.2,
      color: "000000",
      align: "center",
      valign: "middle",
      fit: "shrink"
    });
  }

  function addLiveCard(card, x, y, w) {
    const headH = 0.27;
    const bandH = 0.24;
    const contentY = y + 0.84;
    const contentH = 1.31;

    // 日期标题
    addRect(x, y, w, headH, CC.cardBg, { color: CC.border, width: 0.8 });
    addText(card.dateTitle || "", x, y + 0.02, w, headH - 0.02, {
      fontSize: 8.8,
      color: CC.primary,
      bold: true,
      align: "center",
      valign: "middle"
    });

    // 深色横条
    addRect(x, y + headH, w, bandH, CC.primaryDark);
    addText(`${card.barTitle || ""}      ${card.timeText || ""}`, x, y + headH, w, bandH, {
      fontSize: 8.5,
      color: "FFFFFF",
      bold: true,
      align: "center",
      valign: "middle",
      fit: "shrink"
    });

    // 说明文案
    addText(card.subTitle || "", x, y + 0.58, w, 0.22, {
      fontSize: 8,
      color: CC.primary,
      align: "center",
      valign: "middle",
      fit: "shrink"
    });

    // 内容框
    addRect(x, contentY, w, contentH, CC.cardBg, { color: CC.border, width: 0.8 });
    addText(richItems(card.items || card.content), x + 0.08, contentY + 0.07, w - 0.16, contentH - 0.12, {
      fontSize: 6.45,
      color: "000000",
      align: "left",
      valign: "top",
      lineSpacingMultiple: 0.92,
      fit: "shrink"
    });
  }

  function addVideoBox(content, x, y, w, h) {
    addRect(x, y, w, h, CC.videoBg);
    addText(content || "", x + 0.12, y + 0.10, w - 0.24, h - 0.16, {
      fontSize: 6.7,
      color: "000000",
      align: "center",
      valign: "middle",
      lineSpacingMultiple: 1.1,
      fit: "shrink"
    });
  }

  // 顶部标题
  addText(D.title, 0.37, 0.18, 7.75, 0.34, {
    fontSize: 16,
    bold: true,
    color: CC.textDark,
    align: "left",
    valign: "middle"
  });

  // 右上角 Logo：有图片则放图；无图片则用文字占位
  if (D.logoPath) {
    slide.addImage({
      path: D.logoPath,
      x: 8.35,
      y: 0.16,
      w: 1.25,
      h: 0.34
    });
  } else {
    addText("◆", 8.38, 0.15, 0.28, 0.34, {
      fontSize: 18,
      color: CC.logo,
      bold: true,
      align: "center",
      valign: "middle"
    });
    addText(D.logoText || "", 8.72, 0.17, 1.20, 0.32, {
      fontSize: 14,
      color: CC.logo,
      bold: true,
      align: "left",
      valign: "middle"
    });
  }

  // 顶部细线：青色 + 金色 + 黑色
  addRect(0.37, 0.56, 2.12, 0.012, CC.primary);
  addRect(2.57, 0.56, 1.18, 0.012, CC.accent);
  addRect(3.75, 0.56, 5.90, 0.012, CC.lineDark);

  // 主题横幅
  addRect(0.37, 0.63, 9.36, 0.45, CC.primaryLight);
  addText(D.themeTitle, 0.37, 0.69, 9.36, 0.32, {
    fontSize: 15.5,
    bold: true,
    color: "000000",
    align: "center",
    valign: "middle"
  });

  // 目的 / 策略
  addInfoRow("目的", D.objective, 1.16);
  addInfoRow("策略", D.strategy, 1.46);

  // 直播线区域
  addVerticalLabel(D.liveLabel || "直播线", 0.38, 1.82, 0.42, 2.77);
  addLiveCard(D.cards[0] || {}, 0.86, 1.82, 4.31);
  addLiveCard(D.cards[1] || {}, 5.40, 1.82, 4.31);

  // 福利互动条
  addRect(0.86, 4.20, 8.85, 0.28, CC.primaryLight);
  addText(D.welfare || "", 0.96, 4.20, 8.65, 0.28, {
    fontSize: 8.4,
    color: "000000",
    bold: true,
    align: "center",
    valign: "middle",
    fit: "shrink"
  });

  // 视频线区域
  addVerticalLabel(D.videoLabel || "视频线", 0.38, 4.54, 0.42, 0.97);
  addVideoBox((D.videos || [])[0], 0.86, 4.54, 4.31, 0.97);
  addVideoBox((D.videos || [])[1], 5.40, 4.54, 4.31, 0.97);

  // 底部浅青色收边
  addRect(0, 5.56, 10, 0.05, "D7F4F2");
}
