## T9: 暗底全图封面页（深色背景 + 大标题 + 副标题 + 底部署名条）

适合：投标案封面、品牌宣传首页，需要视觉冲击力的场景。可选配背景图片。

参考效果：深色底色 + 右侧大号主标题 + 英文副标题 + 右下角署名条，左侧可放产品/场景图。

```javascript
function createDarkCoverSlide(pres, C, data) {
  // data = { title, subtitle, englishSub, author, bgImage? }
  const slide = pres.addSlide();
  slide.background = { color: C.dark };

  // 如果有背景图，半透明叠加
  if (data.bgImage) {
    slide.addImage({ path: require("path").resolve(data.bgImage), x: 0, y: 0, w: 10, h: 5.625 });
    // 暗色遮罩保证文字可读
    slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 5.625, fill: { color: C.dark, transparency: 40 } });
  }

  // 顶部主色细条
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });
  // 底部主色细条
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 5.35, w: 10, h: 0.04, fill: { color: C.primary } });

  // 主标题（大号，靠右偏上）
  slide.addText(data.title, {
    x: 4.5, y: 1.2, w: 5.2, h: 1.8,
    fontSize: 42, fontFace: FONT, color: "FFFFFF", bold: true, align: "left", valign: "bottom", margin: 0
  });

  // 英文副标题
  if (data.englishSub) {
    slide.addText(data.englishSub, {
      x: 4.5, y: 3.15, w: 5.2, h: 0.4,
      fontSize: 11, fontFace: FONT, color: C.textLight, align: "left", valign: "middle", margin: 0, charSpacing: 3
    });
  }

  // 中文副标题
  if (data.subtitle) {
    slide.addText(data.subtitle, {
      x: 4.5, y: 3.6, w: 5.2, h: 0.5,
      fontSize: 16, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });
  }

  // 右下署名条
  if (data.author) {
    slide.addShape(pres.shapes.RECTANGLE, {
      x: 6.0, y: 4.5, w: 3.5, h: 0.45, fill: { color: C.dark, transparency: 30 }
    });
    slide.addText(data.author, {
      x: 6.0, y: 4.5, w: 3.2, h: 0.45,
      fontSize: 11, fontFace: FONT, color: C.textLight, align: "right", valign: "middle", margin: 0
    });
    // 小三角装饰
    slide.addShape(pres.shapes.RECTANGLE, { x: 9.25, y: 4.6, w: 0.18, h: 0.18, fill: { color: C.primary }, rotate: 45 });
  }
}
```
