## T28: 用户洞察数据页（左侧画像图表区 + 右侧2×2数据网格，每格有大数字+标题+描述）

适合：用户人群洞察、用户画像分析、市场调研数据展示。左侧放饼图/环形图等画像数据（用色块圆环模拟），右侧2×2网格每格展示一个关键数据维度（大数字+标题+描述）。

```javascript
function createUserInsightSlide(pres, C, data) {
  // data = {
  //   title, summary,
  //   profiles: [{ label, value, color? }],  // 左侧画像数据（如年龄分布、性别分布等），用列表展示
  //   profileTitle,                          // 画像区标题，如"基础画像分布"
  //   grid: [{ icon, title, number, unit, desc }],  // 右侧2×2网格（4个数据格）
  //   sourceNote   // 底部数据来源
  // }
  const slide = pres.addSlide();
  slide.background = { color: C.bg };
  slide.addShape(pres.shapes.RECTANGLE, { x: 0, y: 0, w: 10, h: 0.05, fill: { color: C.primary } });

  // 标题区
  slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.25, w: 0.06, h: 0.45, fill: { color: C.primary } });
  slide.addText(data.title, {
    x: 0.75, y: 0.25, w: 8, h: 0.45,
    fontSize: 20, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 摘要条
  if (data.summary) {
    slide.addShape(pres.shapes.RECTANGLE, { x: 0.5, y: 0.85, w: 9, h: 0.4, fill: { color: C.primaryLight } });
    slide.addText(data.summary, {
      x: 0.7, y: 0.85, w: 8.6, h: 0.4,
      fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "center", valign: "middle", margin: 0
    });
  }

  const contentY = 1.45;
  const contentH = 3.8;

  // ===== 左侧画像区 =====
  const leftW = 3.0;
  slide.addShape(pres.shapes.RECTANGLE, {
    x: 0.5, y: contentY, w: leftW, h: contentH,
    fill: { color: C.cardBg }, shadow: makeShadow()
  });

  // 画像标题
  slide.addText(data.profileTitle || "基础画像", {
    x: 0.7, y: contentY + 0.1, w: leftW - 0.4, h: 0.3,
    fontSize: 12, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
  });

  // 画像数据（用色块+文字列表模拟图表）
  const profiles = data.profiles || [];
  const profStartY = contentY + 0.5;
  const profH = (contentH - 0.7) / Math.max(profiles.length, 1);

  profiles.forEach((p, i) => {
    const py = profStartY + i * profH;
    const barColor = p.color || C.primary;
    // 色块条
    slide.addShape(pres.shapes.ROUNDED_RECTANGLE, {
      x: 0.7, y: py + 0.05, w: 0.5, h: 0.22,
      fill: { color: barColor }, rectRadius: 0.03
    });
    // 标签
    slide.addText(p.label, {
      x: 1.3, y: py, w: 1.5, h: 0.3,
      fontSize: 9, fontFace: FONT, color: C.textDark, align: "left", valign: "middle", margin: 0
    });
    // 数值
    slide.addText(p.value, {
      x: 2.6, y: py, w: 0.7, h: 0.3,
      fontSize: 10, fontFace: FONT, color: C.primary, bold: true, align: "right", valign: "middle", margin: 0
    });
  });

  // ===== 右侧2×2数据网格 =====
  const grid = data.grid || [];
  const gridX = 3.8;
  const gridW = 5.7;
  const cellGap = 0.2;
  const cellW = (gridW - cellGap) / 2;
  const cellH = (contentH - cellGap) / 2;

  grid.forEach((cell, i) => {
    const col = i % 2;
    const row = Math.floor(i / 2);
    const cx = gridX + col * (cellW + cellGap);
    const cy = contentY + row * (cellH + cellGap);

    // 卡片背景
    slide.addShape(pres.shapes.RECTANGLE, {
      x: cx, y: cy, w: cellW, h: cellH,
      fill: { color: C.cardBg }, shadow: makeShadow()
    });

    // 标题
    slide.addText(cell.title || "", {
      x: cx + 0.15, y: cy + 0.1, w: cellW - 0.3, h: 0.3,
      fontSize: 11, fontFace: FONT, color: C.textDark, bold: true, align: "left", valign: "middle", margin: 0
    });

    // 大数字
    const numText = (cell.number || "") + (cell.unit || "");
    slide.addText(numText, {
      x: cx + 0.15, y: cy + 0.45, w: cellW - 0.3, h: 0.6,
      fontSize: 32, fontFace: FONT, color: C.primary, bold: true, align: "left", valign: "middle", margin: 0
    });

    // 描述
    slide.addText(cell.desc || "", {
      x: cx + 0.15, y: cy + 1.1, w: cellW - 0.3, h: cellH - 1.25,
      fontSize: 8, fontFace: FONT, color: C.textMid, align: "left", valign: "top",
      lineSpacingMultiple: 1.4, margin: 0
    });
  });

  // 底部数据来源
  if (data.sourceNote) {
    slide.addText(data.sourceNote, {
      x: 0.5, y: 5.35, w: 9, h: 0.2,
      fontSize: 7, fontFace: FONT, color: C.textLight, align: "left", valign: "middle", margin: 0
    });
  }
}
```
